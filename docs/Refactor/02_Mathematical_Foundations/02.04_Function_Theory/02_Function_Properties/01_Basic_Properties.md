        let mut odd_count = 0;

    all (\(x, y) -> f x <= f y) (zip points (tail points))

isMonotonicDecreasing f points =

        oddCount = length [x | x <- testPoints, x /= 0, abs (f x + f (-x)) < 1e-10]

    in (True, Just minVal, Just maxVal)
