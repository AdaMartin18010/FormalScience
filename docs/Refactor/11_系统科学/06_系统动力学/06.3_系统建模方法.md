## 11.23.2 经典模型

### 11.23.2.1 Bass扩散模型

**定义 11.23.1**（Bass模型）：新产品采纳

$$\frac{dF}{dt} = (p + qF)(1 - F)$$

其中：

- $F$：市场渗透率
- $p$：创新系数
- $q$：模仿系数

### 11.23.2.2 捕食者-猎物模型

**定义 11.23.2**（Lotka-Volterra）：

$$\begin{cases}
\frac{dx}{dt} = \alpha x - \beta xy \\
\frac{dy}{dt} = \delta xy - \gamma y
\end{cases}$$

"""
系统动力学：系统建模方法
经典SD模型实现

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint


class BassDiffusion:
    """Bass扩散模型"""

    def __init__(self, p: float = 0.03, q: float = 0.38, market_size: float = 1000000):
        """
        Args:
            p: 创新系数
            q: 模仿系数
            market_size: 市场规模
        """
        self.p = p
        self.q = q
        self.m = market_size

    def dynamics(self, F, t):
        """Bass方程 dF/dt = (p + qF)(1 - F)"""
        return (self.p + self.q * F) * (1 - F)

    def simulate(self, F0: float = 0, t_span: tuple = (0, 10), n_points: int = 1000):
        """仿真"""
        t = np.linspace(t_span[0], t_span[1], n_points)
        F = odeint(self.dynamics, F0, t).flatten()

        # 计算采纳率和累积采纳
        adoption_rate = np.array([self.dynamics(f, 0) for f in F]) * self.m
        cumulative = F * self.m

        return t, F, adoption_rate, cumulative

    def peak_time(self) -> float:
        """计算销售高峰时间"""
        if self.q > self.p:
            return -1 / (self.p + self.q) * np.log(self.p / self.q)
        return 0


class LotkaVolterra:
    """捕食者-猎物模型"""

    def __init__(self, alpha: float = 1.0, beta: float = 0.1,
                 delta: float = 0.075, gamma: float = 1.5):
        """
        Args:
            alpha: 猎物增长率
            beta: 捕食率
            delta: 捕食者转化效率
            gamma: 捕食者死亡率
        """
        self.alpha = alpha
        self.beta = beta
        self.delta = delta
        self.gamma = gamma

    def dynamics(self, state, t):
        """Lotka-Volterra方程"""
        x, y = state
        dx = self.alpha * x - self.beta * x * y
        dy = self.delta * x * y - self.gamma * y
        return [dx, dy]

    def simulate(self, x0: float, y0: float, t_span: tuple = (0, 50), n_points: int = 2000):
        """仿真"""
        t = np.linspace(t_span[0], t_span[1], n_points)
        trajectory = odeint(self.dynamics, [x0, y0], t)
        return t, trajectory


class InventoryModel:
    """库存管理系统模型"""

    def __init__(self, target_inventory: float = 100,
                 adjustment_time: float = 2,
                 acquisition_delay: float = 3):
        self.target = target_inventory
        self.adjust_time = adjustment_time
        self.delay = acquisition_delay

        # 状态
        self.inventory = target_inventory
        self.goods_in_transit = 0
        self.delay_pipeline = []

    def step(self, demand: float, dt: float):
        """单步仿真"""
        # 订单决策
        inventory_gap = self.target - self.inventory
        desired_acquisition = demand + inventory_gap / self.adjust_time

        # 发出订单（进入延迟管道）
        self.delay_pipeline.append((len(self.delay_pipeline) * dt, desired_acquisition))

        # 处理延迟到达
        received = 0
        new_pipeline = []
        for arrival_time, amount in self.delay_pipeline:
            if len(self.delay_pipeline) * dt - arrival_time >= self.delay:
                received += amount
            else:
                new_pipeline.append((arrival_time, amount))
        self.delay_pipeline = new_pipeline

        # 库存更新
        self.inventory += received - demand

        return self.inventory

    def simulate(self, demand_schedule: list, dt: float = 0.1):
        """运行仿真"""
        inventory_history = [self.inventory]

        for demand in demand_schedule:
            inv = self.step(demand, dt)
            inventory_history.append(inv)

        return inventory_history
