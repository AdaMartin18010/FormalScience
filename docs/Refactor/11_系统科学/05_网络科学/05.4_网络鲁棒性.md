## 11.20.1 网络攻击与容错

### 11.20.1.1 随机故障 vs 蓄意攻击

**定义 11.20.1**（随机故障）：以均匀概率随机移除网络中的节点或边。

**定义 11.20.2**（蓄意攻击）：按度（或介数）降序移除节点。

**定理 11.20.1**（无标度网络的鲁棒性悖论）：

- 对**随机故障**：无标度网络极其鲁棒（故障不影响连通性）
- 对**蓄意攻击**：无标度网络极其脆弱（移除高度节点即崩溃）

### 11.20.1.2 渗流理论

**定义 11.20.3**（渗流阈值）：网络失去巨分量的临界移除比例 $f_c$。

**定理 11.20.2**（ER网络的渗流阈值）：

$$f_c = 1 - \frac{1}{\langle k \rangle}$$

**定理 11.20.3**（无标度网络的鲁棒性）：对于 $\gamma \leq 3$ 的无标度网络：

$$f_c \to 1 \text{ (当 } N \to \infty \text{)}$$

即需要移除几乎所有节点才能摧毁网络。

## 11.20.2 级联失效

### 11.20.2.1 负载-容量模型

**定义 11.20.4**（节点容量）：节点 $i$ 的容量 $C_i = (1 + \alpha) L_i$，其中 $L_i$ 为初始负载，$\alpha$ 为容差参数。

**定义 11.20.5**（级联失效）：节点失效 → 负载重分配 → 新节点过载 → 更多失效。

### 11.20.2.2 网络韧性

**定义 11.20.6**（网络韧性）：网络在扰动后恢复到功能状态的能力。

"""
网络科学：网络鲁棒性
攻击、渗流和级联失效
"""

import numpy as np
import matplotlib.pyplot as plt
import networkx as nx


class NetworkRobustness:
    """网络鲁棒性分析"""

    def __init__(self, G: nx.Graph):
        self.G = G.copy()
        self.n = G.number_of_nodes()
        self.original_G = G.copy()

    def random_attack(self, fraction: float) -> nx.Graph:
        """随机攻击"""
        G_attack = self.G.copy()
        n_remove = int(fraction * self.n)
        nodes_to_remove = np.random.choice(G_attack.nodes(), n_remove, replace=False)
        G_attack.remove_nodes_from(nodes_to_remove)
        return G_attack

    def targeted_attack(self, fraction: float) -> nx.Graph:
        """蓄意攻击（按度排序）"""
        G_attack = self.G.copy()
        n_remove = int(fraction * self.n)

        # 按度排序
        degrees = dict(G_attack.degree())
        nodes_sorted = sorted(degrees.keys(), key=lambda x: degrees[x], reverse=True)
        nodes_to_remove = nodes_sorted[:n_remove]

        G_attack.remove_nodes_from(nodes_to_remove)
        return G_attack

    def giant_component_size(self, G: nx.Graph) -> int:
        """最大连通分量大小"""
        if G.number_of_nodes() == 0:
            return 0
        return len(max(nx.connected_components(G), key=len))

    def average_path_length(self, G: nx.Graph) -> float:
        """平均路径长度"""
        try:
            return nx.average_shortest_path_length(G)
        except:
            return float('inf')

    def analyze_robustness(self, fractions: np.ndarray) -> dict:
        """分析鲁棒性"""
        results = {
            'random': {'gc': [], 'apl': []},
            'targeted': {'gc': [], 'apl': []}
        }

        for f in fractions:
            # 随机攻击
            G_rand = self.random_attack(f)
            results['random']['gc'].append(self.giant_component_size(G_rand) / self.n)

            # 蓄意攻击
            G_targ = self.targeted_attack(f)
            results['targeted']['gc'].append(self.giant_component_size(G_targ) / self.n)

        return results


class CascadingFailure:
    """级联失效模型"""

    def __init__(self, G: nx.Graph, alpha: float = 0.2):
        """
        Args:
            G: 网络
            alpha: 容差参数 C = (1+α)L
        """
        self.G = G.copy()
        self.alpha = alpha
        self.capacity = {}
        self.load = {}
        self._initialize_loads()

    def _initialize_loads(self):
        """初始化负载（基于介数中心性）"""
        betweenness = nx.betweenness_centrality(self.G)
        for node in self.G.nodes():
            self.load[node] = betweenness[node]
            self.capacity[node] = (1 + self.alpha) * self.load[node]

    def cascade(self, initial_failures: list) -> dict:
        """
        模拟级联失效

        Returns:
            失效历史
        """
        G_temp = self.G.copy()
        failed = set(initial_failures)
        history = [len(failed)]

        while True:
            # 移除失效节点
            G_temp.remove_nodes_from(failed)

            if G_temp.number_of_nodes() == 0:
                break

            # 重新计算负载
            betweenness = nx.betweenness_centrality(G_temp)
            new_failures = set()

            for node in G_temp.nodes():
                if node in betweenness:
                    new_load = betweenness[node]
                    # 检查是否过载（与原容量比较）
                    if new_load > self.capacity[node]:
                        new_failures.add(node)

            # 如果没有新的失效，停止
            if len(new_failures) == 0:
                break

            failed.update(new_failures)
            history.append(len(failed))

        return {'total_failed': len(failed), 'history': history}


if __name__ == "__main__":
    fig, axes = plt.subplots(2, 2, figsize=(12, 10))

    # 创建网络
    n = 500
    G_er = nx.erdos_renyi_graph(n, 0.01)
    G_ba = nx.barabasi_albert_graph(n, 3)

    # 1. 随机 vs 蓄意攻击（ER网络）
    ax1 = axes[0, 0]
    robust_er = NetworkRobustness(G_er)
    fractions = np.linspace(0, 0.5, 20)
    results_er = robust_er.analyze_robustness(fractions)

    ax1.plot(fractions, results_er['random']['gc'], 'b-o',
             label='Random attack', linewidth=2)
    ax1.plot(fractions, results_er['targeted']['gc'], 'r-s',
             label='Targeted attack', linewidth=2)
    ax1.set_xlabel('Fraction of nodes removed')
    ax1.set_ylabel('Relative size of giant component')
    ax1.set_title('Erdős-Rényi: Random vs Targeted')
    ax1.legend()
    ax1.grid(True, alpha=0.3)

    # 2. 随机 vs 蓄意攻击（BA网络）
    ax2 = axes[0, 1]
    robust_ba = NetworkRobustness(G_ba)
    results_ba = robust_ba.analyze_robustness(fractions)

    ax2.plot(fractions, results_ba['random']['gc'], 'b-o',
             label='Random attack', linewidth=2)
    ax2.plot(fractions, results_ba['targeted']['gc'], 'r-s',
             label='Targeted attack', linewidth=2)
    ax2.set_xlabel('Fraction of nodes removed')
    ax2.set_ylabel('Relative size of giant component')
    ax2.set_title('Barabási-Albert: Random vs Targeted')
    ax2.legend()
    ax2.grid(True, alpha=0.3)

    # 3. 网络比较
    ax3 = axes[1, 0]
    ax3.plot(fractions, results_er['random']['gc'], 'b--',
             label='ER - Random', linewidth=2)
    ax3.plot(fractions, results_ba['random']['gc'], 'b-',
             label='BA - Random', linewidth=2)
    ax3.plot(fractions, results_er['targeted']['gc'], 'r--',
             label='ER - Targeted', linewidth=2)
    ax3.plot(fractions, results_ba['targeted']['gc'], 'r-',
             label='BA - Targeted', linewidth=2)
    ax3.set_xlabel('Fraction of nodes removed')
    ax3.set_ylabel('Relative size of giant component')
    ax3.set_title('Network Robustness Comparison')
    ax3.legend()
    ax3.grid(True, alpha=0.3)

    # 4. 级联失效
    ax4 = axes[1, 1]

    alphas = [0.1, 0.3, 0.5, 1.0]
    colors = ['red', 'orange', 'green', 'blue']

    for alpha, color in zip(alphas, colors):
        cascade = CascadingFailure(G_ba, alpha=alpha)

        # 移除最高度节点作为初始失效
        degrees = dict(G_ba.degree())
        highest_degree_node = max(degrees.keys(), key=lambda x: degrees[x])

        result = cascade.cascade([highest_degree_node])
        history = result['history']

        ax4.plot(history, 'o-', color=color, label=f'α={alpha}', linewidth=2)

    ax4.set_xlabel('Cascade step')
    ax4.set_ylabel('Cumulative failures')
    ax4.set_title('Cascading Failure (Targeted Initial Attack)')
    ax4.legend()
    ax4.grid(True, alpha=0.3)

    plt.tight_layout()
    plt.savefig('network_robustness.png', dpi=150)
    plt.show()

    # 计算渗流阈值
    print("Network Robustness Analysis:")
    print("-" * 50)

    for name, G in [('Erdős-Rényi', G_er), ('Barabási-Albert', G_ba)]:
        robust = NetworkRobustness(G)

        # 估计渗流阈值（随机攻击）
        f_test = np.linspace(0, 1, 100)
        for i, f in enumerate(f_test):
            G_attack = robust.random_attack(f)
            gc = robust.giant_component_size(G_attack)
            if gc < 0.01 * G.number_of_nodes():
                print(f"{name}: Percolation threshold fc ≈ {f:.3f}")
                break

    print("\nNetwork robustness visualization saved!")
