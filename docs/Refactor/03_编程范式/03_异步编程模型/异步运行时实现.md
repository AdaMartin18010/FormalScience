## Future 与异步状态机

### 自定义 Future trait

```rust
pub trait Future {
    type Output;

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output>;
}

pub enum Poll<T> {
    Ready(T),
    Pending,
}
```

### 基本 Future 实现

```rust
/// 立即完成的 Future
pub struct Ready<T>(Option<T>);

impl<T> Future for Ready<T> {
    type Output = T;

    fn poll(mut self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<T> {
        Poll::Ready(self.0.take().expect("poll called after completion"))
    }
}

/// 挂起指定次数的 Future
pub struct Yield {
    times: usize,
}

impl Future for Yield {
    type Output = ();

    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<()> {
        if self.times > 0 {
            self.times -= 1;
            cx.waker().wake_by_ref();
            Poll::Pending
        } else {
            Poll::Ready(())
        }
    }
}
```

### 组合 Future

```rust
/// Then：顺序执行两个 Future
pub struct Then<A, B, F> {
    first: Option<A>,
    second: Option<B>,
    f: Option<F>,
}

impl<A, B, F> Future for Then<A, B, F>
where
    A: Future,
    B: Future,
    F: FnOnce(A::Output) -> B,
{
    type Output = B::Output;

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        // 先执行 first，完成后执行 second
        // ...
    }
}

/// Join：等待两个 Future 都完成
pub struct Join<A, B> {
    a: Option<A>,
    b: Option<B>,
}

impl<A: Future, B: Future> Future for Join<A, B> {
    type Output = (A::Output, B::Output);

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        // 同时轮询两个 Future
        // ...
    }
}
```

## 任务调度

### 调度器 trait

```rust
pub trait Scheduler {
    fn push(&mut self, task: Task);
    fn pop(&mut self) -> Option<Task>;
    fn wake(&mut self, id: TaskId);
}
```

### FIFO 调度器

```rust
pub struct FifoScheduler {
    ready: VecDeque<Task>,
    blocked: Vec<Task>,
}

impl Scheduler for FifoScheduler {
    fn push(&mut self, task: Task) {
        self.ready.push_back(task);
    }

    fn pop(&mut self) -> Option<Task> {
        self.ready.pop_front()
    }

    fn wake(&mut self, id: TaskId) {
        if let Some(pos) = self.blocked.iter().position(|t| t.id() == id) {
            let mut task = self.blocked.remove(pos);
            task.state = TaskState::Ready;
            self.ready.push_back(task);
        }
    }
}
```

### 优先级调度器

```rust
pub struct PriorityTask {

impl Ord for PriorityTask {
    fn cmp(&self, other: &Self) -> Ordering {
        // 高优先级在前
        other.priority.cmp(&self.priority)
```

### 轮转调度器

```rust
pub struct RoundRobinScheduler {
    ready: VecDeque<Task>,
    time_slice: usize,
    current_ticks: usize,
}

impl RoundRobinScheduler {
    pub fn tick(&mut self) -> bool {
        self.current_ticks += 1;
        if self.current_ticks >= self.time_slice {
            self.current_ticks = 0;
            true // 需要切换任务
        } else {
            false
        }
    }
}
```

### 工作窃取调度器

```rust
pub struct WorkStealingScheduler {
    local_queue: VecDeque<Task>,
    global_queue: Mutex<VecDeque<Task>>,
    stealers: Vec<Mutex<VecDeque<Task>>>,
}

impl WorkStealingScheduler {
    /// 窃取任务
    pub fn steal(&self, worker_id: usize) -> Option<Task> {
        // 尝试从其他工作线程窃取
        for (id, stealer) in self.stealers.iter().enumerate() {
            if id == worker_id { continue; }
            if let Some(task) = stealer.lock().unwrap().pop_back() {
                return Some(task);
            }
        }
        None
    }
}
```

## 执行器

### 基本执行器

```rust
pub struct Executor {
    queue: Arc<Mutex<TaskQueue>>,
    shutdown: AtomicBool,
}

impl Executor {
    pub fn spawn<F>(&self, future: F) -> TaskId
    where
        F: Future<Output = ()> + Send + 'static,
    {
        let task = Task::new(future);
        let id = task.id();

        let mut queue = self.queue.lock().unwrap();
        queue.push(task);

        id
    }

    pub fn run(&self) {
        while !self.shutdown.load(Ordering::SeqCst) {
            if let Some(mut task) = self.next_task() {
                let waker = TaskWaker::new(task.id()).into();
                let mut context = Context::from_waker(&waker);

                match task.poll(&mut context) {
                    Poll::Ready(()) => {}
                    Poll::Pending => {
                        // 重新加入队列
                        self.queue.lock().unwrap().push(task);
                    }
                }
            } else {
                std::thread::yield_now();
            }
        }
    }
}
```

### 定时器

```rust
pub struct Timer {
    deadline: Instant,
}

impl Future for Timer {
    type Output = ();

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<()> {
        if Instant::now() >= self.deadline {
            Poll::Ready(())
        } else {
            cx.waker().wake_by_ref();
            Poll::Pending
        }
    }
}
```
