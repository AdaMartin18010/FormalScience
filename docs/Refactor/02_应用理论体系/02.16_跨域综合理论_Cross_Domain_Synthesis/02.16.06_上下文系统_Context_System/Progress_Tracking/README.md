        let serialized = serde_json::to_string(state)?;
        std::fs::write(&self.state_file, serialized)?;
        Ok(())

        Ok(state)
    }

        let state = self.load_state()?;

        self.continue_work(&state.current_module)?;
        Ok(())
    }
