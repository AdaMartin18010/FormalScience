## 1. 分片策略

### 1.1 分片的基本概念

**定义 17.3.1** (数据分片)：

数据分片是将数据集 $D$ 划分为若干子集 $D_1, D_2, ..., D_n$ 的过程，满足：
$$D = \bigcup_{i=1}^{n} D_i \quad \text{且} \quad \forall i \neq j: D_i \cap D_j = \emptyset \text{ (不相交分片)}$$

**定义 17.3.2** (分片函数)：

分片函数 $f: K \rightarrow P$ 将键 $k \in K$ 映射到分片 $p \in P = \{1, 2, ..., n\}$。

### 1.2 哈希分片

**定义 17.3.3** (哈希分片)：

哈希分片使用哈希函数将数据均匀分布：
$$f(k) = h(k) \mod n$$

其中 $h$ 是哈希函数（如 MurmurHash3, CityHash）。

**优点**：

- 均匀分布：$\forall p_i: |D_i| \approx \frac{|D|}{n}$
- 点查询高效：$O(1)$ 定位分片

**缺点**：

- 范围查询低效：需要扫描所有分片
- 扩容困难：重新分片代价 $O(|D|)$

```python
import hashlib
from typing import List, Dict, Any
import mmh3  # MurmurHash3

class HashSharding:
    """
    哈希分片实现
    """

    def __init__(self, num_shards: int):
        self.num_shards = num_shards

    def get_shard(self, key: str) -> int:
        """
        计算key所属分片

        使用MurmurHash3提供良好的分布性
        """
        hash_value = mmh3.hash(key) & 0x7FFFFFFF  # 取正数
        return hash_value % self.num_shards

    def get_shard_for_batch(self, keys: List[str]) -> Dict[int, List[str]]:
        """批量计算分片归属"""
        result = {i: [] for i in range(self.num_shards)}
        for key in keys:
            shard = self.get_shard(key)
            result[shard].append(key)
        return result

    def add_shard(self) -> int:
        """
        增加分片

        Returns:
            新分片ID
        """
        new_shard_id = self.num_shards
        self.num_shards += 1
        # 注意：需要重新分配数据（代价高）
        return new_shard_id


class ConsistentHashing:
    """
    一致性哈希实现

    解决普通哈希分片扩容时需要重新分配大量数据的问题
    """

    def __init__(self, replicas: int = 150):
        """
        Args:
            replicas: 每个物理节点的虚拟节点数
        """
        self.replicas = replicas
        self.ring = {}  # {hash_value: node_id}
        self.sorted_keys = []
        self.nodes = set()

    def add_node(self, node_id: str):
        """添加节点"""
        self.nodes.add(node_id)
        for i in range(self.replicas):
            # 为每个节点创建多个虚拟节点
            virtual_key = f"{node_id}:{i}"
            hash_val = mmh3.hash(virtual_key) & 0x7FFFFFFF
            self.ring[hash_val] = node_id

        self.sorted_keys = sorted(self.ring.keys())

    def remove_node(self, node_id: str):
        """移除节点"""
        self.nodes.discard(node_id)
        for i in range(self.replicas):
            virtual_key = f"{node_id}:{i}"
            hash_val = mmh3.hash(virtual_key) & 0x7FFFFFFF
            if hash_val in self.ring:
                del self.ring[hash_val]

        self.sorted_keys = sorted(self.ring.keys())

    def get_node(self, key: str) -> str:
        """获取key对应的节点"""
        if not self.ring:
            return None

        hash_val = mmh3.hash(key) & 0x7FFFFFFF

        # 二分查找第一个大于等于hash_val的位置
        idx = self._binary_search(hash_val)

        if idx == len(self.sorted_keys):
            idx = 0

        return self.ring[self.sorted_keys[idx]]

    def _binary_search(self, hash_val: int) -> int:
        """二分查找"""
        left, right = 0, len(self.sorted_keys)

        while left < right:
            mid = (left + right) // 2
            if self.sorted_keys[mid] < hash_val:
                left = mid + 1
            else:
                right = mid

        return left

    def get_affected_keys_on_add(self, new_node: str,
                                  keys: List[str]) -> List[str]:
        """
        计算添加新节点时需要迁移的key

        只有那些映射到新节点的key需要迁移
        """
        affected = []

        # 计算新节点的虚拟节点位置
        new_node_hashes = set()
        for i in range(self.replicas):
            virtual_key = f"{new_node}:{i}"
            hash_val = mmh3.hash(virtual_key) & 0x7FFFFFFF
            new_node_hashes.add(hash_val)

        for key in keys:
            # 检查在新环中的位置是否由新节点服务
            hash_val = mmh3.hash(key) & 0x7FFFFFFF

            # 在原环中的归属
            old_node = self._get_node_without(new_node, hash_val)
            # 在新环中的归属
            new_node_assigned = self.get_node(key)

            if new_node_assigned == new_node and old_node != new_node:
                affected.append(key)

        return affected

    def _get_node_without(self, exclude_node: str, hash_val: int) -> str:
        """计算不包含某节点时的归属"""
        # 简化实现：直接返回下一个节点
        idx = self._binary_search(hash_val)
        if idx >= len(self.sorted_keys):
            idx = 0

        # 跳过被排除的节点
        while self.ring[self.sorted_keys[idx]] == exclude_node:
            idx = (idx + 1) % len(self.sorted_keys)

        return self.ring[self.sorted_keys[idx]]
```

### 1.3 范围分片

**定义 17.3.4** (范围分片)：

范围分片根据键的范围划分数据：
$$f(k) = i \iff b_{i-1} \leq k < b_i$$

其中 $B = \{b_0, b_1, ..., b_n\}$ 是分片边界。

**优点**：

- 高效的范围查询
- 支持前缀扫描
- 易于扩容（分裂分片）

**缺点**：

- 数据倾斜风险（热点）
- 需要定期重新平衡

```python
from bisect import bisect_right
from typing import List, Tuple, Optional

class RangeSharding:
    """
    范围分片实现
    """

    def __init__(self):
        self.boundaries: List[str] = []  # 分片边界
        self.shards: List[str] = []      # 分片ID

    def add_shard(self, shard_id: str, boundary: str):
        """
        添加分片

        Args:
            shard_id: 分片标识
            boundary: 分片上界（不包括）
        """
        # 保持边界有序
        idx = bisect_right(self.boundaries, boundary)
        self.boundaries.insert(idx, boundary)
        self.shards.insert(idx, shard_id)

    def get_shard(self, key: str) -> Optional[str]:
        """获取key所属分片"""
        if not self.boundaries:
            return None

        idx = bisect_right(self.boundaries, key)
        if idx < len(self.shards):
            return self.shards[idx]
        return None

    def get_shard_for_range(self, start_key: str,
                            end_key: str) -> List[str]:
        """获取范围查询涉及的分片"""
        start_idx = bisect_right(self.boundaries, start_key)
        end_idx = bisect_right(self.boundaries, end_key)

        return self.shards[start_idx:end_idx + 1]

    def split_shard(self, shard_id: str, split_key: str,
                    new_shard_id: str):
        """
        分裂分片

        将包含split_key的分片分裂为两个
        """
        idx = self.shards.index(shard_id)

        # 插入新的边界
        self.boundaries.insert(idx, split_key)
        self.shards.insert(idx + 1, new_shard_id)

    def merge_shards(self, shard_id1: str, shard_id2: str):
        """合并相邻分片"""
        # 实现略
        pass
```

### 1.4 列表分片与复合分片

**定义 17.3.5** (列表分片)：

列表分片显式指定键值到分片的映射：
$$f(k) = M[k]$$

其中 $M: K \rightarrow P$ 是离散映射。

**定义 17.3.6** (复合分片)：

复合分片组合多种策略：
$$f(k_1, k_2) = (f_1(k_1), f_2(k_2))$$

例如：先按用户ID哈希，再按时间范围。

```python
class CompositeSharding:
    """
    复合分片实现
    """

    def __init__(self,
                 dimension1: 'ShardingStrategy',
                 dimension2: 'ShardingStrategy'):
        self.dim1 = dimension1
        self.dim2 = dimension2

    def get_shard(self, key1: str, key2: str) -> Tuple[int, int]:
        """计算复合分片位置"""
        shard1 = self.dim1.get_shard(key1)
        shard2 = self.dim2.get_shard(key2)
        return (shard1, shard2)

    def get_flat_shard_id(self, key1: str, key2: str) -> int:
        """
        将复合分片映射为单一分片ID

        用于实际的物理分片定位
        """
        shard1 = self.dim1.get_shard(key1)
        shard2 = self.dim2.get_shard(key2)

        # 假设两个维度分别有n1和n2个分片
        n2 = self.dim2.num_shards
        return shard1 * n2 + shard2


class ShardingStrategy:
    """分片策略接口"""

    def __init__(self, num_shards: int):
        self.num_shards = num_shards

    def get_shard(self, key: str) -> int:
        raise NotImplementedError
```

## 2. 副本调度与一致性

### 2.1 副本模型

**定义 17.3.7** (副本组)：

副本组 $R$ 是数据分片 $D_i$ 的多个副本集合：
$$R(D_i) = \{r_1, r_2, ..., r_k\}$$

其中 $k$ 是副本因子（replication factor）。

**定义 17.3.8** (一致性级别)：

对于读操作，一致性级别 $CL$ 定义为：

- $CL = \text{ONE}$：从任意一个副本读取
- $CL = \text{QUORUM}$：从多数副本读取并比较
- $CL = \text{ALL}$：从所有副本读取

### 2.2 Raft 共识算法

**定义 17.3.9** (Raft)：

Raft 是管理复制日志的共识算法，包含三个子问题：

1. **领导者选举**：选出一个节点作为领导者
2. **日志复制**：领导者将日志条目复制到跟随者
3. **安全性**：保证所有已提交的日志条目一致

**算法 17.3.1** (Raft 领导者选举)：

```python
import random
import asyncio
from enum import Enum
from typing import List, Optional

class RaftState(Enum):
    FOLLOWER = 1
    CANDIDATE = 2
    LEADER = 3

class RaftNode:
    """
    Raft节点实现
    """

    def __init__(self, node_id: str, peers: List[str]):
        self.node_id = node_id
        self.peers = peers
        self.state = RaftState.FOLLOWER

        # 持久化状态
        self.current_term = 0
        self.voted_for = None
        self.log = []  # [(term, command), ...]

        # 易失状态
        self.commit_index = 0
        self.last_applied = 0

        # 领导者状态（仅领导者使用）
        self.next_index = {}   # {peer: next_log_index}
        self.match_index = {}  # {peer: highest_replicated_index}

        # 选举定时器
        self.election_timeout = random.uniform(150, 300)  # ms
        self.heartbeat_interval = 50  # ms

        self.leader_id = None
        self.votes_received = 0

    async def run(self):
        """主循环"""
        while True:
            if self.state == RaftState.FOLLOWER:
                await self._run_follower()
            elif self.state == RaftState.CANDIDATE:
                await self._run_candidate()
            elif self.state == RaftState.LEADER:
                await self._run_leader()

    async def _run_follower(self):
        """跟随者状态"""
        try:
            await asyncio.wait_for(
                self._wait_for_append_entries(),
                timeout=self.election_timeout / 1000
            )
        except asyncio.TimeoutError:
            # 超时，转为候选人
            self._become_candidate()

    async def _run_candidate(self):
        """候选人状态"""
        self.current_term += 1
        self.voted_for = self.node_id
        self.votes_received = 1

        # 向所有节点请求投票
        tasks = [self._request_vote(peer) for peer in self.peers]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        for result in results:
            if isinstance(result, bool) and result:
                self.votes_received += 1

        # 检查是否获得多数
        if self.votes_received > len(self.peers) // 2:
            self._become_leader()

    async def _run_leader(self):
        """领导者状态"""
        # 发送心跳/日志复制
        tasks = [self._send_append_entries(peer) for peer in self.peers]
        await asyncio.gather(*tasks, return_exceptions=True)

        await asyncio.sleep(self.heartbeat_interval / 1000)

    def _become_candidate(self):
        """成为候选人"""
        self.state = RaftState.CANDIDATE
        print(f"{self.node_id} becoming candidate for term {self.current_term + 1}")

    def _become_leader(self):
        """成为领导者"""
        self.state = RaftState.LEADER
        self.leader_id = self.node_id
        print(f"{self.node_id} became leader for term {self.current_term}")

        # 初始化领导者状态
        last_log_index = len(self.log)
        for peer in self.peers:
            self.next_index[peer] = last_log_index + 1
            self.match_index[peer] = 0

    async def _request_vote(self, peer: str) -> bool:
        """向peer请求投票"""
        # 模拟RPC调用
        last_log_index = len(self.log)
        last_log_term = self.log[-1][0] if self.log else 0

        request = {
            'term': self.current_term,
            'candidate_id': self.node_id,
            'last_log_index': last_log_index,
            'last_log_term': last_log_term
        }

        # 实际实现：发送RPC
        response = await self._send_rpc(peer, 'RequestVote', request)

        if response and response.get('vote_granted'):
            return True

        if response and response.get('term') > self.current_term:
            self.current_term = response['term']
            self.state = RaftState.FOLLOWER
            self.voted_for = None

        return False

    async def _send_append_entries(self, peer: str):
        """发送AppendEntries RPC"""
        prev_log_index = self.next_index[peer] - 1
        prev_log_term = self.log[prev_log_index - 1][0] if prev_log_index > 0 else 0

        # 准备要发送的条目
        entries = self.log[prev_log_index:]

        request = {
            'term': self.current_term,
            'leader_id': self.node_id,
            'prev_log_index': prev_log_index,
            'prev_log_term': prev_log_term,
            'entries': entries,
            'leader_commit': self.commit_index
        }

        response = await self._send_rpc(peer, 'AppendEntries', request)

        if response:
            if response.get('term') > self.current_term:
                self.current_term = response['term']
                self.state = RaftState.FOLLOWER
                self.voted_for = None
                return

            if response.get('success'):
                # 更新匹配索引
                self.match_index[peer] = prev_log_index + len(entries)
                self.next_index[peer] = self.match_index[peer] + 1

                # 检查是否可以提交
                self._try_commit()
            else:
                # 日志不匹配，回退next_index
                self.next_index[peer] = max(1, self.next_index[peer] - 1)

    def _try_commit(self):
        """尝试提交日志"""
        for index in range(self.commit_index + 1, len(self.log) + 1):
            # 计算有多少节点复制了此条目
            count = 1  # 领导者自己
            for peer in self.peers:
                if self.match_index.get(peer, 0) >= index:
                    count += 1

            # 如果多数已复制且是当前term，可以提交
            if count > len(self.peers) // 2 and self.log[index - 1][0] == self.current_term:
                self.commit_index = index

    async def _send_rpc(self, peer: str, method: str,
                       request: dict) -> Optional[dict]:
        """发送RPC（模拟）"""
        # 实际实现：通过网络发送
        return None

    def handle_request_vote(self, request: dict) -> dict:
        """处理投票请求"""
        response = {'term': self.current_term, 'vote_granted': False}

        if request['term'] < self.current_term:
            return response

        if request['term'] > self.current_term:
            self.current_term = request['term']
            self.state = RaftState.FOLLOWER
            self.voted_for = None

        # 检查日志是否至少一样新
        last_log_index = len(self.log)
        last_log_term = self.log[-1][0] if self.log else 0

        log_ok = (request['last_log_term'] > last_log_term or
                  (request['last_log_term'] == last_log_term and
                   request['last_log_index'] >= last_log_index))

        if (self.voted_for is None or self.voted_for == request['candidate_id']) and log_ok:
            self.voted_for = request['candidate_id']
            response['vote_granted'] = True

        return response

    def handle_append_entries(self, request: dict) -> dict:
        """处理AppendEntries请求"""
        response = {'term': self.current_term, 'success': False}

        if request['term'] < self.current_term:
            return response

        # 重置选举定时器
        self.state = RaftState.FOLLOWER
        self.leader_id = request['leader_id']

        if request['term'] > self.current_term:
            self.current_term = request['term']
            self.voted_for = None

        # 检查日志一致性
        prev_log_index = request['prev_log_index']
        if prev_log_index > len(self.log):
            return response

        if prev_log_index > 0:
            if self.log[prev_log_index - 1][0] != request['prev_log_term']:
                return response

        # 追加新条目
        entries = request['entries']
        for i, (term, command) in enumerate(entries):
            index = prev_log_index + i
            if index < len(self.log):
                if self.log[index][0] != term:
                    # 删除冲突条目及之后所有条目
                    self.log = self.log[:index]
                    self.log.append((term, command))
            else:
                self.log.append((term, command))

        # 更新commit_index
        if request['leader_commit'] > self.commit_index:
            self.commit_index = min(request['leader_commit'], len(self.log))

        response['success'] = True
        return response
```

```python
from enum import Enum
import asyncio
from typing import List, Optional, Any

class ConsistencyLevel(Enum):
    ONE = 1
    TWO = 2
    THREE = 3
    QUORUM = 4
    ALL = 5
    LOCAL_QUORUM = 6
    EACH_QUORUM = 7

class ReplicatedStorage:
    """
    副本存储实现
    """

    def __init__(self, replication_factor: int = 3):
        self.replication_factor = replication_factor
        self.nodes = {}  # {node_id: Node}
        self.ring = ConsistentHashing()

    async def read(self, key: str,
                   consistency: ConsistencyLevel) -> Optional[Any]:
        """
        读取数据

        根据一致性级别从不同数量的副本读取
        """
        replicas = self._get_replicas(key)
        required_reads = self._required_reads(consistency)

        # 并行读取
        tasks = [self._read_from_node(node, key) for node in replicas]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 收集成功的响应
        successful = [r for r in results if not isinstance(r, Exception) and r is not None]

        if len(successful) < required_reads:
            raise Exception(f"Not enough replicas available: {len(successful)} < {required_reads}")

        # 返回最新版本（基于时间戳）
        if consistency in [ConsistencyLevel.QUORUM, ConsistencyLevel.ALL]:
            # 比较并返回最新
            return self._resolve_latest(successful)

        return successful[0]

    async def write(self, key: str, value: Any,
                    consistency: ConsistencyLevel) -> bool:
        """
        写入数据
        """
        replicas = self._get_replicas(key)
        required_writes = self._required_writes(consistency)

        timestamp = time.time()

        # 并行写入
        tasks = [self._write_to_node(node, key, value, timestamp)
                 for node in replicas]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        successful = sum(1 for r in results if not isinstance(r, Exception) and r)

        return successful >= required_writes

    def _get_replicas(self, key: str, n: int = None) -> List[str]:
        """获取key的副本节点列表"""
        n = n or self.replication_factor
        replicas = []

        # 使用一致性哈希获取节点
        # 简化：返回前n个不同节点
        for i in range(n):
            virtual_key = f"{key}:{i}"
            node = self.ring.get_node(virtual_key)
            if node not in replicas:
                replicas.append(node)

        return replicas

    def _required_reads(self, consistency: ConsistencyLevel) -> int:
        """计算读操作需要的副本数"""
        if consistency == ConsistencyLevel.ONE:
            return 1
        elif consistency == ConsistencyLevel.QUORUM:
            return (self.replication_factor // 2) + 1
        elif consistency == ConsistencyLevel.ALL:
            return self.replication_factor
        elif consistency == ConsistencyLevel.TWO:
            return min(2, self.replication_factor)
        elif consistency == ConsistencyLevel.THREE:
            return min(3, self.replication_factor)
        return 1

    def _required_writes(self, consistency: ConsistencyLevel) -> int:
        """计算写操作需要的副本数"""
        return self._required_reads(consistency)

    def _resolve_latest(self, results: List[tuple]) -> Any:
        """从多个副本结果中返回最新版本"""
        # 假设结果是 (value, timestamp) 元组
        return max(results, key=lambda x: x[1])[0]

    async def _read_from_node(self, node_id: str, key: str):
        """从指定节点读取"""
        # 实际实现：RPC调用
        pass

    async def _write_to_node(self, node_id: str, key: str,
                            value: Any, timestamp: float):
        """写入指定节点"""
```

```python
class CrossDCReplication:
    """
    跨数据中心复制
    """

    def __init__(self, mode: str = 'leader_follower'):
        self.mode = mode
        self.leader_dc = None
        self.followers = []
        self.sync_replicas = 1  # 同步复制的副本数

    async def write(self, key: str, value: Any) -> bool:
        """跨数据中心写入"""
        if self.mode == 'leader_follower':
            return await self._leader_follower_write(key, value)
        elif self.mode == 'multi_master':
            return await self._multi_master_write(key, value)
        elif self.mode == 'consensus':
            return await self._consensus_write(key, value)

    async def _leader_follower_write(self, key: str, value: Any) -> bool:
        """主从模式写入"""
        # 1. 写入主数据中心
        success = await self._write_to_dc(self.leader_dc, key, value)
        if not success:
            return False

        # 2. 异步复制到从数据中心
        replication_tasks = [
            self._async_replicate(dc, key, value)
            for dc in self.followers
        ]
        asyncio.create_task(asyncio.gather(*replication_tasks,
                                           return_exceptions=True))

        return True

    async def _multi_master_write(self, key: str, value: Any) -> bool:
        """多主模式写入"""
        # 使用向量时钟处理冲突
        vector_clock = self._get_vector_clock(key)
        vector_clock[self.leader_dc] = vector_clock.get(self.leader_dc, 0) + 1

        # 广播到其他主节点
        tasks = [
            self._write_with_vclock(dc, key, value, vector_clock)
            for dc in self.followers
        ]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 检查是否有冲突
        conflicts = [r for r in results if isinstance(r, ConflictException)]
        if conflicts:
            # 冲突解决
            await self._resolve_conflict(key, value, vector_clock)

        return True

    async def _consensus_write(self, key: str, value: Any) -> bool:
        """共识模式写入（跨DC Paxos）"""
        # 使用Paxos在多个数据中心间达成共识
        all_dcs = [self.leader_dc] + self.followers

        # Phase 1: Prepare
        promises = []
        for dc in all_dcs:
            promise = await self._send_prepare(dc, key, value)
            if promise:
                promises.append(promise)

        if len(promises) <= len(all_dcs) // 2:
            return False

        # Phase 2: Accept
        accepts = 0
        for dc in all_dcs:
            accepted = await self._send_accept(dc, key, value)
            if accepted:
                accepts += 1

        return accepts > len(all_dcs) // 2
```

## 4. 负载均衡

### 4.1 负载均衡算法

**定义 17.3.12** (负载均衡问题)：

给定请求集合 $R$ 和节点集合 $N$，负载均衡目标是：
$$\min \max_{n \in N} L(n) \quad \text{s.t.} \quad \sum_{n} L(n) = \sum_{r \in R} w(r)$$

其中 $L(n)$ 是节点 $n$ 的负载，$w(r)$ 是请求 $r$ 的权重。

**定义 17.3.13** (加权轮询)：

节点 $n_i$ 的权重为 $w_i$，则选中概率为：
$$P(n_i) = \frac{w_i}{\sum_{j} w_j}$$

```python
import random
import hashlib
from typing import List, Dict, Callable
from dataclasses import dataclass
import heapq

@dataclass
class Node:
    """后端节点"""
    id: str
    weight: int = 1
    current_connections: int = 0
    current_load: float = 0.0
    healthy: bool = True
    response_time_ms: float = 0.0

class LoadBalancer:
    """
    负载均衡器基类
    """

    def __init__(self, nodes: List[Node]):
        self.nodes = nodes
        self.healthy_nodes = nodes

    def select_node(self, key: str = None) -> Node:
        """选择节点"""
        raise NotImplementedError

    def update_node_health(self, node_id: str, healthy: bool):
        """更新节点健康状态"""
        for node in self.nodes:
            if node.id == node_id:
                node.healthy = healthy
                break

        self.healthy_nodes = [n for n in self.nodes if n.healthy]


class WeightedRoundRobin(LoadBalancer):
    """
    加权轮询负载均衡

    平滑加权轮询算法（Nginx实现）
    """

    def __init__(self, nodes: List[Node]):
        super().__init__(nodes)
        self.current_weight = 0
        self.last_selected = -1
        self.gcd_weight = self._gcd([n.weight for n in nodes])
        self.max_weight = max(n.weight for n in nodes)

    def select_node(self, key: str = None) -> Node:
        """选择下一个节点"""
        while True:
            self.last_selected = (self.last_selected + 1) % len(self.healthy_nodes)

            if self.last_selected == 0:
                self.current_weight -= self.gcd_weight
                if self.current_weight <= 0:
                    self.current_weight = self.max_weight
                    if self.current_weight == 0:
                        return None

            node = self.healthy_nodes[self.last_selected]
            if node.weight >= self.current_weight:
                return node

    def _gcd(self, weights: List[int]) -> int:
        """计算最大公约数"""
        from math import gcd
        from functools import reduce
        return reduce(gcd, weights)


class ConsistentHashBalancer(LoadBalancer):
    """
    一致性哈希负载均衡

    保证相同key总是路由到相同节点
    """

    def __init__(self, nodes: List[Node], virtual_nodes: int = 150):
        super().__init__(nodes)
        self.virtual_nodes = virtual_nodes
        self.ring = {}
        self._build_ring()

    def _build_ring(self):
        """构建哈希环"""
        for node in self.nodes:
            for i in range(self.virtual_nodes):
                virtual_key = f"{node.id}#{i}"
                hash_val = self._hash(virtual_key)
                self.ring[hash_val] = node

    def select_node(self, key: str) -> Node:
        """根据key选择节点"""
        if not self.ring:
            return None

        hash_val = self._hash(key)

        # 找到第一个大于等于hash_val的节点
        sorted_hashes = sorted(self.ring.keys())
        for h in sorted_hashes:
            if h >= hash_val and self.ring[h].healthy:
                return self.ring[h]

        # 回环到第一个节点
        return self.ring[sorted_hashes[0]]

    def _hash(self, key: str) -> int:
        """计算哈希值"""
        return int(hashlib.md5(key.encode()).hexdigest(), 16)


class LeastConnections(LoadBalancer):
    """
    最少连接数负载均衡
    """

    def select_node(self, key: str = None) -> Node:
        """选择连接数最少的节点"""
        if not self.healthy_nodes:
            return None

        return min(self.healthy_nodes, key=lambda n: n.current_connections)


class AdaptiveLoadBalancer(LoadBalancer):
    """
    自适应负载均衡

    综合考虑响应时间、连接数、权重
    """

    def select_node(self, key: str = None) -> Node:
        """
        基于多因素评分选择节点

        评分 = weight / (response_time * connections + 1)
        """
        if not self.healthy_nodes:
            return None

        best_node = None
        best_score = -1

        for node in self.healthy_nodes:
            # 负载因子
            load_factor = (node.response_time_ms + 1) * (node.current_connections + 1)
            score = node.weight / load_factor

            if score > best_score:
                best_score = score
                best_node = node

        return best_node
```

```python
class RequestRouter:
    """
    请求路由器

    支持读写分离、强制主库等策略
    """

    def __init__(self):
        self.master_node = None
        self.replica_nodes = []
        self.read_consistency = 'eventual'  # eventual|session|strong

    def route(self, request_type: str, consistency: str = None) -> Node:
        """
        路由请求到合适的节点

        Args:
            request_type: 'read' | 'write'
            consistency: 'strong' | 'session' | 'eventual'
        """
        if request_type == 'write':
            return self.master_node

        # 读请求
        consistency = consistency or self.read_consistency

        if consistency == 'strong':
            # 强一致性读，走主库
            return self.master_node
        elif consistency == 'session':
            # 会话一致性，需要跟踪写入位置
            return self._route_session_consistent()
        else:
            # 最终一致性，读副本
            return self._route_to_replica()

    def _route_session_consistent(self) -> Node:
        """会话一致性路由"""
        # 如果本会话有写入，需要读取包含写入的副本
        # 简化实现
        return self._route_to_replica()

    def _route_to_replica(self) -> Node:
        """路由到副本"""
        # 使用最少连接策略
        if not self.replica_nodes:
            return self.master_node

        return min(self.replica_nodes, key=lambda n: n.current_connections)
```

## 5. 故障恢复调度

### 5.1 故障检测

**定义 17.3.14** (故障模型)：

| 故障类型 | 描述 | 检测方法 |
|---------|------|---------|
| 崩溃停止 (Crash-Stop) | 节点永久失效 | 心跳超时 |
| 崩溃恢复 (Crash-Recovery) | 节点暂时失效后恢复 | 心跳+日志 |
| 拜占庭故障 (Byzantine) | 节点行为任意 | 多数投票 |
| 网络分区 (Network Partition) | 网络分裂 | 仲裁机制 |

**定义 17.3.15** (故障检测器)：

故障检测器是一个函数 $D: N \times T \rightarrow \{Suspect, Trust\}$，其中：

- $N$ 是节点集合
- $T$ 是时间

```python
import asyncio
import time
from enum import Enum
from typing import Dict, Callable, Optional
from dataclasses import dataclass

class NodeStatus(Enum):
    HEALTHY = 1
    SUSPECTED = 2
    FAILED = 3

@dataclass
class Heartbeat:
    node_id: str
    timestamp: float
    sequence_number: int

class FailureDetector:
    """
    故障检测器实现

    基于Phi累积故障检测器
    """

    def __init__(self, threshold: float = 8.0,
                 min_std_deviation_ms: float = 500):
        self.threshold = threshold
        self.min_std_deviation = min_std_deviation_ms

        self.heartbeats: Dict[str, List[float]] = {}  # 到达间隔历史
        self.last_heartbeat: Dict[str, float] = {}
        self.status: Dict[str, NodeStatus] = {}

        self.failure_callbacks: List[Callable] = []
        self.suspect_callbacks: List[Callable] = []

    def heartbeat_received(self, node_id: str):
        """收到心跳"""
        now = time.time() * 1000  # 毫秒

        if node_id in self.last_heartbeat:
            interval = now - self.last_heartbeat[node_id]
            if node_id not in self.heartbeats:
                self.heartbeats[node_id] = []
            self.heartbeats[node_id].append(interval)

            # 只保留最近100个样本
            if len(self.heartbeats[node_id]) > 100:
                self.heartbeats[node_id] = self.heartbeats[node_id][-100:]

        self.last_heartbeat[node_id] = now

        # 恢复状态
        if self.status.get(node_id) != NodeStatus.HEALTHY:
            self.status[node_id] = NodeStatus.HEALTHY

    def get_phi(self, node_id: str) -> float:
        """
        计算Phi值（怀疑程度）

        基于正态分布假设
        """
        if node_id not in self.heartbeats or not self.heartbeats[node_id]:
            return float('inf')

        intervals = self.heartbeats[node_id]
        mean = sum(intervals) / len(intervals)
        variance = sum((x - mean) ** 2 for x in intervals) / len(intervals)
        std_dev = max(variance ** 0.5, self.min_std_deviation)

        last = self.last_heartbeat.get(node_id, 0)
        time_since_last = time.time() * 1000 - last

        # Phi = -log10(F(t))
        # F(t) 是CDF：P(X <= t)
        y = (time_since_last - mean) / std_dev

        # 近似计算
        phi = -1 * self._log10_cdf_approx(y)

        return phi

    def _log10_cdf_approx(self, y: float) -> float:
        """CDF的近似计算"""
        import math
        # 使用误差函数近似
        cdf = 0.5 * (1 + math.erf(y / math.sqrt(2)))
        if cdf <= 0:
            return -10
        return math.log10(cdf)

    async def monitor(self):
        """监控循环"""
        while True:
            for node_id in self.last_heartbeat:
                phi = self.get_phi(node_id)

                if phi > self.threshold:
                    # 超过阈值，标记为失败
                    if self.status.get(node_id) != NodeStatus.FAILED:
                        self.status[node_id] = NodeStatus.FAILED
                        for callback in self.failure_callbacks:
                            callback(node_id)
                elif phi > self.threshold * 0.5:
                    # 可疑状态
                    if self.status.get(node_id) == NodeStatus.HEALTHY:
                        self.status[node_id] = NodeStatus.SUSPECTED
                        for callback in self.suspect_callbacks:
                            callback(node_id)

            await asyncio.sleep(1)

    def on_failure(self, callback: Callable):
        """注册故障回调"""
        self.failure_callbacks.append(callback)

    def on_suspect(self, callback: Callable):
        """注册怀疑回调"""
        self.suspect_callbacks.append(callback)
```

### 5.2 自动故障转移

```python
class FailoverManager:
    """
    故障转移管理器
    """

    def __init__(self):
        self.leader = None
        self.followers = []
        self.election_in_progress = False

    async def handle_leader_failure(self, leader_id: str):
        """处理主节点故障"""
        if self.election_in_progress:
            return

        self.election_in_progress = True

        try:
            # 1. 确认主节点确实故障（避免误判）
            if not await self._confirm_failure(leader_id):
                return

            # 2. 选举新主节点
            new_leader = await self._elect_new_leader()

            # 3. 提升新主节点
            await self._promote_leader(new_leader)

            # 4. 更新路由
            await self._update_routing(new_leader)

            # 5. 通知客户端
            await self._notify_clients(new_leader)

        finally:
            self.election_in_progress = False

    async def _elect_new_leader(self) -> str:
        """
        选举新主节点

        使用优先级：
        1. 数据最新的节点
        2. 延迟最低的节点
        3. ID最小的节点
        """
        candidates = []

        for follower in self.followers:
```

Spanner 是全球分布的 NewSQL 数据库，提供：

**TrueTime API**：

- $TT.now()$: 返回时间区间 $[earliest, latest]$
- $TT.after(t)$: 若 $t$ 已过去则返回 true
- $TT.before(t)$: 若 $t$ 未到来则返回 true

**定理 17.3.1** (Spanner 外部一致性)：

若事务 $T_2$ 在 $T_1$ 提交后开始，则：
$$commit(T_1) < start(T_2) \Rightarrow ts(T_1) < ts(T_2)$$

### 6.2 CockroachDB 架构

**定义 17.3.17** (CockroachDB)：

CockroachDB 是开源的分布式 SQL 数据库，特点：

- **混合逻辑时钟 (HLC)**：结合物理时钟和逻辑时钟
- **租约持有者 (LeaseHolder)**：每个Range有一个租约持有者处理读写
- **事务流水线**：优化延迟

```python
class CockroachDBLike:
    """
    类似CockroachDB的架构概念
    """

    class HybridLogicalClock:
        """混合逻辑时钟"""

        def __init__(self):
            self.wall_time = 0  # 物理时间（纳秒）
            self.logical = 0    # 逻辑计数器

        def now(self) -> tuple:
            """获取当前HLC时间戳"""
            physical = int(time.time() * 1e9)

            if physical > self.wall_time:
                self.wall_time = physical
                self.logical = 0
            else:
                self.logical += 1

            return (self.wall_time, self.logical)

        def update(self, received: tuple):
            """根据收到的时钟更新本地时钟"""
            recv_wall, recv_logical = received

            if recv_wall > self.wall_time:
                self.wall_time = recv_wall
                self.logical = recv_logical + 1
            elif recv_wall == self.wall_time:
                self.logical = max(self.logical, recv_logical) + 1
            else:
                self.logical += 1

    class Range:
        """数据分片（Range）"""

        def __init__(self, start_key: str, end_key: str):
            self.start_key = start_key
            self.end_key = end_key
            self.replicas = []  # 副本节点
            self.lease_holder = None  # 租约持有者
```

```rust
        let mut ready = self.raft_group.ready_since(self.last_applying_idx);

        // 1. 发送消息给其他节点
        self.send_ready_messages(ctx, &ready);

        self.raft_group.propose(vec![], data)?;

            index: self.raft_group.raft.raft_log.last_index(),
            term: self.raft_group.raft.term,
```

```go
// MVCCGet 实现MVCC读取
func MVCCGet(
    ctx context.Context,
    reader Reader,
    key roachpb.Key,
    timestamp hlc.Timestamp,
    opts MVCCGetOptions,
) (*roachpb.Value, *roachpb.Intent, error) {

    // 1. 检查是否有intent（未提交的事务写入）
    metaKey := MakeMVCCMetadataKey(key)
    meta := &MVCCMetadata{}

    ok, _, _, err := reader.MVCCGetProto(metaKey, meta)
    if err != nil {
        return nil, nil, err
    }

    if ok && meta.Txn != nil {
        // 发现未提交的intent
        if opts.Inconsistent {
            // 不一致读：忽略intent
        } else {
            // 返回intent给调用者处理
            intent := roachpb.Intent{
                Span:   roachpb.Span{Key: key},
                Txn:    *meta.Txn,
                Status: roachpb.PENDING,
            }
            return nil, &intent, nil
        }
    }

    // 2. 查找可见的最新版本
    // 从timestamp开始向前扫描
    iter := reader.NewMVCCIterator(MVCCKeyAndIntentsIterKind)
    defer iter.Close()

    seekKey := MVCCKey{Key: key, Timestamp: timestamp}
    iter.SeekLT(seekKey)

    if iter.Valid() && iter.UnsafeKey().Key.Equal(key) {
        value := &roachpb.Value{}
        value.RawBytes = iter.UnsafeValue()
        value.Timestamp = iter.UnsafeKey().Timestamp.ToLegacyTimestamp()
        return value, nil, nil
    }

    return nil, nil, nil
}

// MVCCPut 实现MVCC写入
func MVCCPut(
    ctx context.Context,
    rw ReadWriter,
    ms *enginepb.MVCCStats,
    key roachpb.Key,
    timestamp hlc.Timestamp,
    value roachpb.Value,
    opts MVCCWriteOptions,
) error {

    // 1. 读取现有元数据
    metaKey := MakeMVCCMetadataKey(key)
    meta := &MVCCMetadata{}

    ok, origMetaSize, _, err := rw.MVCCGetProto(metaKey, meta)
    if err != nil {
        return err
    }

    // 2. 检查写冲突
    if ok && meta.Txn != nil {
        // 已有未提交的写入
        if opts.Txn == nil || meta.Txn.ID != opts.Txn.ID {
            return &roachpb.WriteIntentError{Intents: []roachpb.Intent{{
                Span:   roachpb.Span{Key: key},
                Txn:    *meta.Txn,
                Status: roachpb.PENDING,
            }}}
        }
    }

    // 3. 检查写太旧（WriteTooOld）
    if meta.Timestamp.ToTimestamp().Greater(timestamp) {
        // 在timestamp之后已有写入
        if opts.Txn == nil {
            return &roachpb.WriteTooOldError{
                Timestamp:       timestamp,
                ActualTimestamp: meta.Timestamp.ToTimestamp(),
            }
        }
        // 事务内需要重试
    }

    // 4. 写入数据
    mvccKey := MVCCKey{Key: key, Timestamp: timestamp}
    if err := rw.PutMVCC(mvccKey, value.RawBytes); err != nil {
        return err
    }

    // 5. 更新元数据（如果是intent）
    if opts.Txn != nil {
        meta.Txn = opts.Txn
        meta.Timestamp = timestamp.ToLegacyTimestamp()
        if err := rw.PutProto(metaKey, meta); err != nil {
            return err
        }
    }

    return nil
```

```go
// TxnCoordSender 是客户端事务协调器
type TxnCoordSender struct {
    // 事务状态
    mu struct {
        syncutil.Mutex
        txnsState txnState
        txnProto  roachpb.Transaction
    }

    // 拦截器链
    interceptors []txnInterceptor

    // 到各Range的连接缓存
    replicas *rangeCache
}

// Send 发送请求
func (tc *TxnCoordSender) Send(
    ctx context.Context,
    ba roachpb.BatchRequest,
) (*roachpb.BatchResponse, *roachpb.Error) {

    tc.mu.Lock()
    defer tc.mu.Unlock()

    // 添加事务信息到请求
    ba.Txn = &tc.mu.txnProto

    // 通过拦截器链处理
    var br *roachpb.BatchResponse
    var pErr *roachpb.Error

    for _, interceptor := range tc.interceptors {
        br, pErr = interceptor.SendLocked(ctx, ba)
        if pErr != nil {
            // 处理错误，可能需要重试
            if err := tc.handleErrLocked(ctx, pErr); err != nil {
                return nil, roachpb.NewError(err)
            }
            // 重试
            continue
        }
    }

    return br, nil
}

}

func (tp *txnPipeliner) SendLocked(

    ba roachpb.BatchRequest,
) (*roachpb.BatchResponse, *roachpb.Error) {

        tp.outstandingWrites.insert(w)
```
