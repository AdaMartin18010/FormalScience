## 1. 内存管理策略

### 1.1 内存分配策略

**定义 17.4.1** (内存分配器)：

内存分配器是一个函数 $A: \mathbb{N} \rightarrow \text{Addr} \cup \{\bot\}$，满足：
$$A(n) = \begin{cases} \text{addr} & \text{if } \exists \text{连续空闲内存块 } \geq n \\ \bot & \text{otherwise} \end{cases}$$

**定义 17.4.2** (内存碎片)：

内部碎片：分配大小 $s$ > 请求大小 $n$：
$$I_{internal} = s - n$$

外部碎片：空闲内存总量 $\geq n$，但无连续块可用：
$$I_{external} = \sum_{i} f_i \text{ where } \forall i: f_i < n \land \sum f_i \geq n$$

```cpp
/**
 * 内存池分配器实现
 *
 * 使用Slab分配策略减少内存碎片
 */

#include <cstddef>
#include <cstdint>
#include <vector>
#include <memory>
#include <mutex>

namespace inmem {

// 内存块大小类别
constexpr size_t SIZE_CLASSES[] = {
    8, 16, 32, 64, 128, 256, 512, 1024,
    2048, 4096, 8192, 16384, 32768, 65536
};
constexpr size_t NUM_SIZE_CLASSES = sizeof(SIZE_CLASSES) / sizeof(SIZE_CLASSES[0]);

/**
 * Slab - 固定大小内存块的管理
 */
class Slab {
public:
    Slab(size_t block_size, size_t num_blocks)
        : block_size_(block_size)
        , num_blocks_(num_blocks)
        , free_list_(nullptr) {

        // 分配连续内存区域
        memory_ = std::make_unique<char[]>(block_size * num_blocks);

        // 初始化空闲列表（使用指针内嵌技术）
        for (size_t i = 0; i < num_blocks; ++i) {
            void* block = memory_.get() + i * block_size;
            *reinterpret_cast<void**>(block) = free_list_;
            free_list_ = block;
        }
    }

    void* Allocate() {
        if (free_list_ == nullptr) {
            return nullptr;  // Slab已满
        }

        void* result = free_list_;
        free_list_ = *reinterpret_cast<void**>(free_list_);
        return result;
    }

    void Deallocate(void* ptr) {
        if (ptr == nullptr) return;

        // 验证ptr是否属于此Slab
        if (ptr < memory_.get() ||
            ptr >= memory_.get() + block_size_ * num_blocks_) {
            return;  // 不属于此Slab
        }

        *reinterpret_cast<void**>(ptr) = free_list_;
        free_list_ = ptr;
    }

    bool Contains(void* ptr) const {
        return ptr >= memory_.get() &&
               ptr < memory_.get() + block_size_ * num_blocks_;
    }

    bool IsEmpty() const {
        return free_list_ == nullptr;
    }

private:
    size_t block_size_;
    size_t num_blocks_;
    std::unique_ptr<char[]> memory_;
    void* free_list_;
};

/**
 * Slab分配器
 */
class SlabAllocator {
public:
    SlabAllocator(size_t blocks_per_slab = 1024)
        : blocks_per_slab_(blocks_per_slab) {

        slabs_.resize(NUM_SIZE_CLASSES);
    }

    void* Allocate(size_t size) {
        // 找到合适的大小类别
        size_t class_idx = GetSizeClass(size);
        if (class_idx >= NUM_SIZE_CLASSES) {
            // 大对象直接使用系统分配
            return ::operator new(size);
        }

        std::lock_guard<std::mutex> lock(mutexes_[class_idx]);

        // 查找有空闲块的Slab
        for (auto& slab : slabs_[class_idx]) {
            void* ptr = slab->Allocate();
            if (ptr != nullptr) {
                return ptr;
            }
        }

        // 所有Slab都满，创建新Slab
        size_t block_size = SIZE_CLASSES[class_idx];
        auto new_slab = std::make_unique<Slab>(block_size, blocks_per_slab_);
        void* ptr = new_slab->Allocate();
        slabs_[class_idx].push_back(std::move(new_slab));

        return ptr;
    }

    void Deallocate(void* ptr, size_t size) {
        if (ptr == nullptr) return;

        size_t class_idx = GetSizeClass(size);
        if (class_idx >= NUM_SIZE_CLASSES) {
            ::operator delete(ptr);
            return;
        }

        std::lock_guard<std::mutex> lock(mutexes_[class_idx]);

        // 找到包含ptr的Slab
        for (auto& slab : slabs_[class_idx]) {
            if (slab->Contains(ptr)) {
                slab->Deallocate(ptr);
                return;
            }
        }
    }

    size_t GetAllocatedSize(size_t size) {
        size_t class_idx = GetSizeClass(size);
        if (class_idx >= NUM_SIZE_CLASSES) {
            return size;
        }
        return SIZE_CLASSES[class_idx];
    }

private:
    size_t GetSizeClass(size_t size) {
        for (size_t i = 0; i < NUM_SIZE_CLASSES; ++i) {
            if (SIZE_CLASSES[i] >= size) {
                return i;
            }
        }
        return NUM_SIZE_CLASSES;  // 大对象
    }

    size_t blocks_per_slab_;
    std::vector<std::vector<std::unique_ptr<Slab>>> slabs_;
    std::mutex mutexes_[NUM_SIZE_CLASSES];
};

} // namespace inmem
```

### 1.2 内存预分配与预留

```python
import mmap
import os
from typing import Optional, List

class MemoryArena:
    """
    内存竞技场 - 批量分配和一次性释放

    适用于事务或查询执行期间的大量小对象分配
    """

    def __init__(self, block_size: int = 64 * 1024 * 1024):  # 64MB
        self.block_size = block_size
        self.blocks: List[bytearray] = []
        self.current_block: Optional[bytearray] = None
        self.current_offset = 0
        self.total_allocated = 0

    def allocate(self, size: int, alignment: int = 8) -> int:
        """
        从竞技场分配内存

        Returns:
            分配的内存偏移量（相对于当前块）
        """
        # 对齐
        aligned_size = (size + alignment - 1) & ~(alignment - 1)

        # 检查当前块是否有足够空间
        if (self.current_block is None or
            self.current_offset + aligned_size > len(self.current_block)):
            # 分配新块
            self._allocate_new_block(max(aligned_size, self.block_size))

        offset = self.current_offset
        self.current_offset += aligned_size
        self.total_allocated += aligned_size

        return offset

    def _allocate_new_block(self, min_size: int):
        """分配新内存块"""
        # 使用mmap获取大块内存
        size = max(min_size, self.block_size)
        new_block = bytearray(size)

        self.blocks.append(new_block)
        self.current_block = new_block
        self.current_offset = 0

    def reset(self):
        """重置竞技场，释放所有分配"""
        self.blocks.clear()
        self.current_block = None
        self.current_offset = 0
        self.total_allocated = 0

    def get_memory(self, block_idx: int, offset: int) -> memoryview:
        """获取指定位置的内存视图"""
        return memoryview(self.blocks[block_idx])[offset:]


class PreallocatedPool:
    """
    预分配对象池

    减少运行时分配开销
    """

    def __init__(self, object_factory, initial_size: int = 1000):
        self.factory = object_factory
        self.available = []
        self.in_use = set()

        # 预分配
        for _ in range(initial_size):
            obj = self.factory()
            self.available.append(obj)

    def acquire(self):
        """获取对象"""
        if not self.available:
            # 扩展池
            for _ in range(len(self.available) + 100):
                self.available.append(self.factory())

        obj = self.available.pop()
```

### 1.3 内存压缩与回收

**定义 17.4.3** (内存压缩)：

内存压缩是重新整理内存布局以减少碎片的过程：
$$\text{Compact}: M \rightarrow M' \text{ where } \text{frag}(M') < \text{frag}(M)$$

**定理 17.4.1** (压缩停止条件)：

内存压缩在达到以下条件时停止：
$$\frac{\text{空闲内存}}{\text{总内存}} < \theta_{low} \lor \text{最大连续块} > n_{min}$$

```python
import gc
import sys
from typing import Dict, List, Callable
import weakref

class MemoryManager:
    """
    内存管理器 - 监控和回收内存
    """

    def __init__(self, max_memory_mb: int = 1024):
        self.max_memory_bytes = max_memory_mb * 1024 * 1024
        self.used_memory = 0
        self.objects: Dict[int, 'TrackedObject'] = {}
        self.gc_threshold = 0.8  # 80%触发GC
        self.compress_threshold = 0.9  # 90%触发压缩

    def allocate(self, size: int,
                 deleter: Callable = None) -> 'MemoryHandle':
        """分配内存"""
        # 检查是否需要回收
        if self.used_memory + size > self.max_memory_bytes * self.gc_threshold:
            self._trigger_gc()

        if self.used_memory + size > self.max_memory_bytes * self.compress_threshold:
            self._trigger_compression()

        if self.used_memory + size > self.max_memory_bytes:
            raise MemoryError("Out of memory")

        self.used_memory += size
        handle = MemoryHandle(size, deleter)
        return handle

    def _trigger_gc(self):
        """触发垃圾回收"""
        # 淘汰过期缓存
        self._evict_expired()

        # Python GC
        gc.collect()

    def _trigger_compression(self):
        """触发内存压缩"""
        # 合并小对象
        self._compact_small_objects()

        # 序列化不活跃数据
        self._serialize_inactive()

    def _evict_expired(self):
        """淘汰过期数据"""
        expired = []
        for obj_id, obj in list(self.objects.items()):
            if obj.is_expired():
                expired.append(obj_id)

        for obj_id in expired:
            self._free_object(obj_id)

    def _compact_small_objects(self):

        pass

    def _serialize_inactive(self):
        """序列化不活跃数据到磁盘"""
        for obj in self.objects.values():
            if obj.inactive_time > 60:  # 60秒不活跃
                obj.serialize_to_disk()

    def _free_object(self, obj_id: int):

        if obj_id in self.objects:

            self.used_memory -= obj.size
            if obj.deleter:
                obj.deleter()
```

## 2. 数据结构优化

### 2.1 自适应基数树 (ART)

**定义 17.4.4** (ART - Adaptive Radix Tree)：

ART 是优化的前缀树，使用自适应节点大小：

- Node4：4个子节点指针
- Node16：16个子节点指针
- Node48：48个子节点指针
- Node256：256个子节点指针

**定理 17.4.2** (ART 空间复杂度)：

对于 $n$ 个键，每个键平均长度 $k$，ART 的空间复杂度为：
$$O(n \cdot k \cdot \lceil \frac{\Sigma}{w} \rceil)$$

其中 $\Sigma$ 是字符集大小，$w$ 是字长。

```cpp
/**
 * 自适应基数树 (ART) 实现
 *
 * 参考："The Adaptive Radix Tree: ARTful Indexing for Main-Memory Databases"
 */

#include <cstdint>
#include <cstring>
#include <memory>
#include <algorithm>

namespace art {

using Key = std::vector<uint8_t>;
using Value = uint64_t;

// 节点类型
enum class NodeType : uint8_t {
    NODE4 = 0,
    NODE16 = 1,
    NODE48 = 2,
    NODE256 = 3,
    LEAF = 4
};

// 基础节点结构
struct Node {
    NodeType type;
    uint8_t num_children;
    uint32_t prefix_length;
    uint8_t prefix[8];  // 压缩路径前缀

    Node(NodeType t) : type(t), num_children(0), prefix_length(0) {
        memset(prefix, 0, sizeof(prefix));
    }

    virtual ~Node() = default;

    bool IsFull() const {
        switch (type) {
            case NodeType::NODE4: return num_children == 4;
            case NodeType::NODE16: return num_children == 16;
            case NodeType::NODE48: return num_children == 48;
            case NodeType::NODE256: return num_children == 256;
            default: return false;
        }
    }

    virtual Node* FindChild(uint8_t key_byte) = 0;
    virtual void AddChild(uint8_t key_byte, Node* child) = 0;
    virtual void RemoveChild(uint8_t key_byte) = 0;
};

// Node4: 最多4个子节点，使用数组存储
struct Node4 : public Node {
    uint8_t keys[4];
    Node* children[4];

    Node4() : Node(NodeType::NODE4) {
        memset(keys, 0, sizeof(keys));
        memset(children, 0, sizeof(children));
    }

    Node* FindChild(uint8_t key_byte) override {
        for (uint8_t i = 0; i < num_children; ++i) {
            if (keys[i] == key_byte) {
                return children[i];
            }
        }
        return nullptr;
    }

    void AddChild(uint8_t key_byte, Node* child) override {
        uint8_t pos = 0;
        while (pos < num_children && keys[pos] < key_byte) {
            ++pos;
        }
        // 移动插入
        memmove(&keys[pos + 1], &keys[pos], num_children - pos);
        memmove(&children[pos + 1], &children[pos],
                (num_children - pos) * sizeof(Node*));
        keys[pos] = key_byte;
        children[pos] = child;
        ++num_children;
    }

    void RemoveChild(uint8_t key_byte) override {
        for (uint8_t i = 0; i < num_children; ++i) {
            if (keys[i] == key_byte) {
                memmove(&keys[i], &keys[i + 1], num_children - i - 1);
                memmove(&children[i], &children[i + 1],
                        (num_children - i - 1) * sizeof(Node*));
                --num_children;
                return;
            }
        }
    }
};

// Node16: 最多16个子节点，使用SIMD优化查找
struct Node16 : public Node {
    uint8_t keys[16];
    Node* children[16];

    Node16() : Node(NodeType::NODE16) {
        memset(keys, 0xFF, sizeof(keys));  // 初始化为无效值
        memset(children, 0, sizeof(children));
    }

    Node* FindChild(uint8_t key_byte) override {
        #ifdef __SSE2__
            // 使用SSE2 SIMD指令批量比较
            __m128i cmp = _mm_cmpeq_epi8(
                _mm_set1_epi8(key_byte),
                _mm_loadu_si128(reinterpret_cast<__m128i*>(keys))
            );
            int mask = _mm_movemask_epi8(cmp);
            if (mask != 0) {
                return children[__builtin_ctz(mask)];
            }
        #else
            // 回退到线性搜索
            for (uint8_t i = 0; i < num_children; ++i) {
                if (keys[i] == key_byte) {
                    return children[i];
                }
            }
        #endif
        return nullptr;
    }

    void AddChild(uint8_t key_byte, Node* child) override {
        // 找到插入位置保持有序
        uint8_t pos = 0;
        while (pos < num_children && keys[pos] < key_byte) {
            ++pos;
        }
        memmove(&keys[pos + 1], &keys[pos], num_children - pos);
        memmove(&children[pos + 1], &children[pos],
                (num_children - pos) * sizeof(Node*));
        keys[pos] = key_byte;
        children[pos] = child;
        ++num_children;
    }

    void RemoveChild(uint8_t key_byte) override {
        for (uint8_t i = 0; i < num_children; ++i) {
            if (keys[i] == key_byte) {
                memmove(&keys[i], &keys[i + 1], num_children - i - 1);
                memmove(&children[i], &children[i + 1],
                        (num_children - i - 1) * sizeof(Node*));
                --num_children;
                return;
            }
        }
    }
};

// Node48: 最多48个子节点，使用索引数组
struct Node48 : public Node {
    uint8_t child_index[256];  // 从key byte到children索引的映射
    Node* children[48];

    Node48() : Node(NodeType::NODE48) {
        memset(child_index, 255, sizeof(child_index));  // 255表示空槽
        memset(children, 0, sizeof(children));
    }

    Node* FindChild(uint8_t key_byte) override {
        uint8_t index = child_index[key_byte];
        if (index != 255) {
            return children[index];
        }
        return nullptr;
    }

    void AddChild(uint8_t key_byte, Node* child) override {
        // 找到第一个空槽
        uint8_t index = 0;
        while (children[index] != nullptr) {
            ++index;
        }
        child_index[key_byte] = index;
        children[index] = child;
        ++num_children;
    }

    void RemoveChild(uint8_t key_byte) override {
        uint8_t index = child_index[key_byte];
        if (index != 255) {
            children[index] = nullptr;
            child_index[key_byte] = 255;
            --num_children;
        }
    }
};

// Node256: 最多256个子节点，直接索引
struct Node256 : public Node {
    Node* children[256];

    Node256() : Node(NodeType::NODE256) {
        memset(children, 0, sizeof(children));
    }

    Node* FindChild(uint8_t key_byte) override {
        return children[key_byte];
    }

    void AddChild(uint8_t key_byte, Node* child) override {
        children[key_byte] = child;
        ++num_children;
    }

    void RemoveChild(uint8_t key_byte) override {
        children[key_byte] = nullptr;
        --num_children;
    }
};

// 叶子节点
struct Leaf : public Node {
    Key key;
    Value value;

    Leaf(const Key& k, Value v)
        : Node(NodeType::LEAF), key(k), value(v) {}
};

/**
 * ART 树
 */
class AdaptiveRadixTree {
public:
    AdaptiveRadixTree() : root_(nullptr) {}

    ~AdaptiveRadixTree() {
        DeleteTree(root_);
    }

    void Insert(const Key& key, Value value) {
        if (!root_) {
            root_ = new Leaf(key, value);
            return;
        }

        InsertHelper(root_, key, 0, value);
    }

    Value* Lookup(const Key& key) {
        Node* node = root_;
        uint32_t depth = 0;

        while (node) {
            if (node->type == NodeType::LEAF) {
                Leaf* leaf = static_cast<Leaf*>(node);
                if (leaf->key == key) {
                    return &leaf->value;
                }
                return nullptr;
            }

            // 检查压缩前缀
            if (depth < key.size()) {
                uint8_t key_byte = key[depth];
                node = node->FindChild(key_byte);
                ++depth;
            } else {
                return nullptr;
            }
        }

        return nullptr;
    }

private:
    void InsertHelper(Node*& node, const Key& key,
                      uint32_t depth, Value value) {
        if (!node) {
            node = new Leaf(key, value);
            return;
        }

        if (node->type == NodeType::LEAF) {
            Leaf* leaf = static_cast<Leaf*>(node);
            if (leaf->key == key) {
                leaf->value = value;  // 更新
                return;
            }

            // 分裂叶子节点
            Node4* new_node = new Node4();

            // 找到公共前缀
            uint32_t common_prefix = 0;
            while (common_prefix < key.size() &&
                   common_prefix < leaf->key.size() &&
                   key[common_prefix] == leaf->key[common_prefix]) {
                ++common_prefix;
            }

            // 设置压缩前缀
            new_node->prefix_length = std::min<uint32_t>(
                common_prefix - depth, 8
            );
            memcpy(new_node->prefix, &key[depth], new_node->prefix_length);

            // 添加原叶子和新叶子
            new_node->AddChild(leaf->key[common_prefix], leaf);
            InsertHelper(new_node, key, common_prefix, value);

            node = new_node;
            return;
        }

        // 内部节点
        if (node->IsFull()) {
            node = GrowNode(node);
        }

        uint8_t key_byte = key[depth];
        Node* child = node->FindChild(key_byte);

        if (!child) {
            // 创建新叶子
            Leaf* new_leaf = new Leaf(key, value);
            node->AddChild(key_byte, new_leaf);
        } else {
            InsertHelper(child, key, depth + 1, value);
        }
    }

    Node* GrowNode(Node* node) {
        // 节点扩容
        switch (node->type) {
            case NodeType::NODE4: {
                Node4* n4 = static_cast<Node4*>(node);
                Node16* n16 = new Node16();
                memcpy(n16->keys, n4->keys, 4);
                memcpy(n16->children, n4->children, 4 * sizeof(Node*));
                n16->num_children = n4->num_children;
                delete n4;
                return n16;
            }
            case NodeType::NODE16: {
                Node16* n16 = static_cast<Node16*>(node);
                Node48* n48 = new Node48();
                for (uint8_t i = 0; i < n16->num_children; ++i) {
                    n48->child_index[n16->keys[i]] = i;
                    n48->children[i] = n16->children[i];
                }
                n48->num_children = n16->num_children;
                delete n16;
                return n48;
            }
            case NodeType::NODE48: {
                Node48* n48 = static_cast<Node48*>(node);
                Node256* n256 = new Node256();
                for (int i = 0; i < 256; ++i) {
                    if (n48->child_index[i] != 255) {
                        n256->children[i] = n48->children[n48->child_index[i]];
                    }
                }
                n256->num_children = n48->num_children;
                delete n48;
                return n256;
            }
            default:
                return node;
        }
    }

    void DeleteTree(Node* node) {
        if (!node) return;

        if (node->type == NodeType::LEAF) {
            delete node;
            return;
        }

        // 递归删除子节点
        switch (node->type) {
            case NodeType::NODE4: {
                Node4* n4 = static_cast<Node4*>(node);
                for (uint8_t i = 0; i < n4->num_children; ++i) {
                    DeleteTree(n4->children[i]);
                }
                break;
            }
            case NodeType::NODE16: {
                Node16* n16 = static_cast<Node16*>(node);
                for (uint8_t i = 0; i < n16->num_children; ++i) {
                    DeleteTree(n16->children[i]);
                }
                break;
            }
            case NodeType::NODE48: {
                Node48* n48 = static_cast<Node48*>(node);
                for (int i = 0; i < 256; ++i) {
                    if (n48->child_index[i] != 255) {
                        DeleteTree(n48->children[n48->child_index[i]]);
                    }
                }
                break;
            }
            case NodeType::NODE256: {
                Node256* n256 = static_cast<Node256*>(node);
                for (int i = 0; i < 256; ++i) {
                    DeleteTree(n256->children[i]);
                }
                break;
            }
            default:
                break;
        }

        delete node;
    }

    Node* root_;
};

} // namespace art
```

### 2.2 跳表 (Skip List)

**定义 17.4.5** (跳表)：

跳表是概率性平衡数据结构，期望搜索复杂度 $O(\log n)$：

$$E[\text{搜索代价}] = \sum_{i=0}^{h} \frac{n}{2^i} = O(\log n)$$

**定理 17.4.3** (跳表高度)：

对于 $n$ 个元素的跳表，以高概率高度为 $O(\log n)$：
$$P(\text{height} > c \log n) \leq \frac{1}{n^{c-1}}$$

```cpp
/**
 * 无锁跳表实现
 */

#include <atomic>
#include <memory>
#include <random>

template<typename K, typename V>
class SkipList {
public:
    static constexpr int MAX_HEIGHT = 16;

    struct Node {
        K key;
        V value;
        std::atomic<int> height;
        std::atomic<Node*> next[MAX_HEIGHT];

        Node(const K& k, const V& v, int h)
            : key(k), value(v), height(h) {
            for (int i = 0; i < MAX_HEIGHT; ++i) {
                next[i].store(nullptr, std::memory_order_relaxed);
            }
        }
    };

    SkipList() : size_(0), height_(1) {
        // 创建头节点
        head_ = new Node(K{}, V{}, MAX_HEIGHT);
    }

    ~SkipList() {
        Node* curr = head_->next[0].load();
        while (curr) {
            Node* next = curr->next[0].load();
            delete curr;
            curr = next;
        }
        delete head_;
    }

    bool Insert(const K& key, const V& value) {
        Node* preds[MAX_HEIGHT];
        Node* succs[MAX_HEIGHT];

        while (true) {
            // 查找位置
            if (FindNode(key, preds, succs)) {
                return false;  // 已存在
            }

            // 创建新节点
            int top_layer = RandomHeight();
            Node* new_node = new Node(key, value, top_layer);

            // 设置指针
            for (int i = 0; i < top_layer; ++i) {
                new_node->next[i].store(succs[i], std::memory_order_relaxed);
            }

            // 尝试链接
            Node* pred = preds[0];
            Node* succ = succs[0];

            if (!pred->next[0].compare_exchange_strong(
                    succ, new_node,
                    std::memory_order_release,
                    std::memory_order_relaxed)) {
                delete new_node;
                continue;  // 重试
            }

            // 链接高层
            for (int i = 1; i < top_layer; ++i) {
                while (true) {
                    pred = preds[i];
                    succ = succs[i];

                    if (pred->next[i].compare_exchange_strong(
                            succ, new_node,
                            std::memory_order_release,
                            std::memory_order_relaxed)) {
                        break;
                    }

                    // 重新查找
                    FindNode(key, preds, succs);
                }
            }

            size_.fetch_add(1, std::memory_order_relaxed);
            UpdateHeight(top_layer);
            return true;
        }
    }

    bool Find(const K& key, V& value) {
        Node* preds[MAX_HEIGHT];
        Node* succs[MAX_HEIGHT];

        if (FindNode(key, preds, succs)) {
            value = succs[0]->value;
            return true;
        }
        return false;
    }

    bool Delete(const K& key) {
        Node* preds[MAX_HEIGHT];
        Node* succs[MAX_HEIGHT];
        Node* node_to_delete = nullptr;

        while (true) {
            if (!FindNode(key, preds, succs)) {
                return false;  // 不存在
            }

            node_to_delete = succs[0];

            // 标记删除（逻辑删除）
            // 简化实现，实际使用更复杂的并发删除
            for (int i = node_to_delete->height - 1; i >= 0; --i) {
                Node* succ = node_to_delete->next[i].load();
                preds[i]->next[i].compare_exchange_strong(
                    node_to_delete, succ,
                    std::memory_order_release,
                    std::memory_order_relaxed
                );
            }

            size_.fetch_sub(1, std::memory_order_relaxed);
            // 注意：实际无锁实现需要延迟回收
            return true;
        }
    }

private:
    bool FindNode(const K& key, Node** preds, Node** succs) {
        bool found = false;
        Node* pred = head_;

        for (int i = height_.load(std::memory_order_acquire) - 1; i >= 0; --i) {
            Node* curr = pred->next[i].load(std::memory_order_acquire);

            while (curr && curr->key < key) {
                pred = curr;
                curr = curr->next[i].load(std::memory_order_acquire);
            }

            preds[i] = pred;
            succs[i] = curr;

            if (curr && curr->key == key) {
                found = true;
            }
        }

        return found;
    }

    int RandomHeight() {
        static thread_local std::mt19937 gen(std::random_device{}());
        std::uniform_real_distribution<> dis(0, 1);

        int height = 1;
        while (height < MAX_HEIGHT && dis(gen) < 0.5) {
            ++height;
        }
        return height;
    }

    void UpdateHeight(int new_height) {
        int expected = height_.load();
        while (new_height > expected) {
            if (height_.compare_exchange_weak(expected, new_height)) {
                break;
            }
        }
    }

    Node* head_;
    std::atomic<size_t> size_;
    std::atomic<int> height_;
};
```

### 2.3 B+树优化

**定义 17.4.6** (内存B+树)：

内存优化B+树特征：

- 节点大小匹配CPU缓存行 ($64$ 字节)
- 使用指针内嵌技术
- 二分搜索优化

```cpp
/**
 * 内存优化B+树
 */

#include <array>
#include <algorithm>
#include <cstring>

constexpr size_t CACHE_LINE_SIZE = 64;
constexpr size_t NODE_SIZE = CACHE_LINE_SIZE * 8;  // 512 bytes

/**
 * 压缩B+树节点
 *
 * 使用SIMD友好的布局
 */
template<typename K, typename V, size_t B = 16>
class BPlusTree {
public:
    struct alignas(CACHE_LINE_SIZE) Node {
        bool is_leaf;
        uint16_t num_keys;

        union {
            struct {
                K keys[B - 1];
                V values[B - 1];
            } leaf;

            struct {
                K keys[B - 1];
                Node* children[B];
            } internal;
        };

        Node* next;  // 叶子节点链表（用于范围扫描）

        Node(bool leaf) : is_leaf(leaf), num_keys(0), next(nullptr) {}
    };

    V* Find(const K& key) {
        Node* node = root_;

        while (!node->is_leaf) {
            // SIMD优化搜索
            int idx = UpperBoundInternal(node, key);
            node = node->internal.children[idx];
        }

        // 叶子节点线性搜索（适合小节点）
        for (uint16_t i = 0; i < node->num_keys; ++i) {
            if (node->leaf.keys[i] == key) {
                return &node->leaf.values[i];
            }
        }

        return nullptr;
    }

    void Insert(const K& key, const V& value) {
        // 根节点分裂处理
        if (root_->num_keys == B - 1) {
            Node* new_root = new Node(false);
            new_root->internal.children[0] = root_;
            SplitChild(new_root, 0, root_);
            root_ = new_root;
        }

        InsertNonFull(root_, key, value);
    }

    // 范围扫描优化
    void RangeScan(const K& start, const K& end,
                   std::vector<std::pair<K, V>>& result) {
        Node* node = FindLeaf(start);

        while (node) {
            for (uint16_t i = 0; i < node->num_keys; ++i) {
                if (node->leaf.keys[i] >= start && node->leaf.keys[i] <= end) {
                    result.emplace_back(node->leaf.keys[i],
                                       node->leaf.values[i]);
                }
                if (node->leaf.keys[i] > end) {
                    return;
                }
            }
            node = node->next;
        }
    }

private:
    // 使用SIMD指令优化的二分搜索
    int UpperBoundInternal(Node* node, const K& key) {
        #ifdef __SSE4_2__
            // SSE4.2 字符串比较优化
            // 实际实现取决于K的类型
        #endif

        // 标准二分搜索
        int left = 0, right = node->num_keys;
        while (left < right) {
            int mid = (left + right) >> 1;
            if (node->internal.keys[mid] <= key) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }

    Node* FindLeaf(const K& key) {
        Node* node = root_;
        while (!node->is_leaf) {
            int idx = UpperBoundInternal(node, key);
            node = node->internal.children[idx];
        }
        return node;
    }

    void SplitChild(Node* parent, int idx, Node* child) {
        Node* new_child = new Node(child->is_leaf);
        new_child->num_keys = (B - 1) / 2;

        // 复制后半部分
        int mid = (B - 1) / 2;
        if (child->is_leaf) {
            memcpy(new_child->leaf.keys, &child->leaf.keys[mid],
                   new_child->num_keys * sizeof(K));
            memcpy(new_child->leaf.values, &child->leaf.values[mid],
                   new_child->num_keys * sizeof(V));
            new_child->next = child->next;
            child->next = new_child;
        } else {
            memcpy(new_child->internal.keys, &child->internal.keys[mid],
                   new_child->num_keys * sizeof(K));
            memcpy(new_child->internal.children, &child->internal.children[mid + 1],
                   (new_child->num_keys + 1) * sizeof(Node*));
        }

        child->num_keys = mid;

        // 插入父节点
        for (int i = parent->num_keys; i > idx; --i) {
            parent->internal.children[i + 1] = parent->internal.children[i];
        }
        parent->internal.children[idx + 1] = new_child;

        for (int i = parent->num_keys - 1; i >= idx; --i) {
            parent->internal.keys[i + 1] = parent->internal.keys[i];
        }
        parent->internal.keys[idx] = child->is_leaf ?
            child->leaf.keys[mid] : child->internal.keys[mid];
        parent->num_keys++;
    }

    void InsertNonFull(Node* node, const K& key, const V& value) {
        int i = node->num_keys - 1;

        if (node->is_leaf) {
            // 找到插入位置
            while (i >= 0 && node->leaf.keys[i] > key) {
                node->leaf.keys[i + 1] = node->leaf.keys[i];
                node->leaf.values[i + 1] = node->leaf.values[i];
                --i;
            }

            node->leaf.keys[i + 1] = key;
            node->leaf.values[i + 1] = value;
            node->num_keys++;
        } else {
            // 找到子节点
            while (i >= 0 && node->internal.keys[i] > key) {
                --i;
            }
            ++i;

            if (node->internal.children[i]->num_keys == B - 1) {
                SplitChild(node, i, node->internal.children[i]);
                if (key > node->internal.keys[i]) {
                    ++i;
                }
            }

            InsertNonFull(node->internal.children[i], key, value);
        }
    }

    Node* root_;
};
```

## 3. 并发控制优化

### 3.1 无锁数据结构

**定义 17.4.7** (无锁进展保证)：

算法是无锁的 (lock-free)，如果：
$$\forall \text{执行}, \exists \text{有限步内某线程完成操作}$$

**定义 17.4.8** (无等待进展保证)：

算法是无等待的 (wait-free)，如果：
$$\forall \text{线程}, \exists \text{有限步} k: \text{操作在} k \text{步内完成}$$

```cpp
/**
 * 无锁队列 (Michael-Scott Queue)
 */

template<typename T>
class LockFreeQueue {
public:
    struct Node {
        std::atomic<T*> data;
        std::atomic<Node*> next;

        Node() : data(nullptr), next(nullptr) {}
    };

    LockFreeQueue() {
        Node* dummy = new Node();
        head_.store(dummy, std::memory_order_relaxed);
        tail_.store(dummy, std::memory_order_relaxed);
    }

    void Enqueue(T* value) {
        Node* new_node = new Node();
        new_node->data.store(value, std::memory_order_relaxed);

        Node* tail;
        Node* next;

        while (true) {
            tail = tail_.load(std::memory_order_acquire);
            next = tail->next.load(std::memory_order_acquire);

            if (tail == tail_.load(std::memory_order_acquire)) {
                if (next == nullptr) {
                    // 尝试链接新节点
                    if (tail->next.compare_exchange_weak(
                            next, new_node,
                            std::memory_order_release,
                            std::memory_order_relaxed)) {
                        break;
                    }
                } else {
                    // 帮助推进tail
                    tail_.compare_exchange_weak(
                        tail, next,
                        std::memory_order_release,
                        std::memory_order_relaxed
                    );
                }
            }
        }

        // 尝试更新tail
        tail_.compare_exchange_strong(
            tail, new_node,
            std::memory_order_release,
            std::memory_order_relaxed
        );
    }

    T* Dequeue() {
        Node* head;
        Node* tail;
        Node* next;
        T* value;

        while (true) {
            head = head_.load(std::memory_order_acquire);
            tail = tail_.load(std::memory_order_acquire);
            next = head->next.load(std::memory_order_acquire);

            if (head == head_.load(std::memory_order_acquire)) {
                if (head == tail) {
                    if (next == nullptr) {
                        return nullptr;  // 空队列
                    }
                    // 帮助推进tail
                    tail_.compare_exchange_weak(
                        tail, next,
                        std::memory_order_release,
                        std::memory_order_relaxed
                    );
                } else {
                    value = next->data.load(std::memory_order_relaxed);
                    if (head_.compare_exchange_weak(
                            head, next,
                            std::memory_order_release,
                            std::memory_order_relaxed)) {
                        break;
                    }
                }
            }
        }

        // 安全回收head
        // 实际实现需要延迟回收（Hazard Pointers）
        return value;
    }

private:
    std::atomic<Node*> head_;
    std::atomic<Node*> tail_;
};
```

### 3.2 乐观锁与读优化

```cpp
/**
 * 乐观锁并发控制
 *
 * 适合读多写少场景
 */

template<typename T>
class OptimisticLock {
public:
    struct VersionedData {
        std::atomic<uint64_t> version;
        T data;
    };

    bool Read(VersionedData* vd, T& result) {
        uint64_t version_before = vd->version.load(std::memory_order_acquire);

        if (version_before & 1) {
            return false;  // 正在写入
        }

        // 读取数据
        result = vd->data;

        // 验证版本
        uint64_t version_after = vd->version.load(std::memory_order_acquire);

        return version_before == version_after;
    }

    bool Write(VersionedData* vd, const T& new_value) {
        uint64_t version = vd->version.load(std::memory_order_relaxed);

        // 尝试获取写锁（设置最低位）
        if (!vd->version.compare_exchange_strong(
                version, version + 1,
                std::memory_order_acquire,
                std::memory_order_relaxed)) {
            return false;
        }

        // 写入数据
        vd->data = new_value;

        // 释放锁并增加版本
        vd->version.store(version + 2, std::memory_order_release);

        return true;
    }
};
```

### 4.1 预写日志 (WAL)

**定义 17.4.9** (WAL)：

预写日志确保先写日志后写数据：
$$\forall \text{更新}: \text{WAL}(update) \prec \text{Apply}(update)$$

**定理 17.4.4** (WAL 恢复正确性)：

使用WAL的数据库在崩溃后可以恢复到一致状态。

**证明概要**：

1. 已提交的日志条目必然已持久化
2. 恢复时重放所有已提交但未应用的日志
3. 未提交的日志可以回滚
4. 因此数据库状态与崩溃前已提交事务一致

```cpp
        size_t aligned_size = ((entry_size + 511) / 512) * 512;

        std::unique_ptr<uint8_t[]> buffer(new uint8_t[aligned_size]);
        memset(buffer.get(), 0, aligned_size);

        entry->lsn = ++current_lsn_;
        entry->data_length = length;

            return 0;
```

### 4.2 快照持久化

```python
import threading
import pickle
from typing import Dict, Any

class SnapshotManager:
    """
    快照管理器

    支持增量快照和全量快照
    """

    def __init__(self):
        self.last_snapshot_lsn = 0
        self.snapshot_in_progress = False
        self.snapshot_lock = threading.Lock()

    def create_full_snapshot(self, data: Dict[str, Any],
                             filename: str) -> bool:
        """创建全量快照"""
        with self.snapshot_lock:
            if self.snapshot_in_progress:
                return False
            self.snapshot_in_progress = True

        try:
            # 序列化数据
            with open(filename, 'wb') as f:
                pickle.dump(data, f, protocol=pickle.HIGHEST_PROTOCOL)

            return True
        finally:
            self.snapshot_in_progress = False

    def create_incremental_snapshot(self,
                                     data: Dict[str, Any],
                                     changes_since_last: Dict[str, Any],
                                     filename: str) -> bool:
        """创建增量快照"""
        with self.snapshot_lock:
            if self.snapshot_in_progress:
                return False
            self.snapshot_in_progress = True

        try:
            snapshot = {
                'base_lsn': self.last_snapshot_lsn,
                'changes': changes_since_last
            }

            with open(filename, 'wb') as f:
                pickle.dump(snapshot, f)

            return True
        finally:
            self.snapshot_in_progress = False

    def load_snapshot(self, filename: str) -> Dict[str, Any]:
        """加载快照"""
        with open(filename, 'rb') as f:
            return pickle.load(f)
```

### 4.3 AOF与RDB对比

| 特性 | AOF (Append-Only File) | RDB (Redis Database) |
|-----|----------------------|---------------------|

| 数据安全 | 更高（可配置fsync） | 可能丢失快照间数据 |

| 恢复速度 | 慢（需重放） | 快（直接加载） |

## 5. 混合存储调度

### 5.1 冷热数据分离

**定义 17.4.10** (数据温度)：

数据温度基于访问频率定义：
$$T(d) = \frac{\sum_{t \in \Delta t} \text{access}(d, t)}{|\Delta t|}$$

**定义 17.4.11** (冷热分离)：

分区策略：

- 热区：$T(d) \geq \theta_{hot}$，驻留内存
- 温区：$\theta_{cold} < T(d) < \theta_{hot}$，可交换到SSD
- 冷区：$T(d) \leq \theta_{cold}$，归档到磁盘

```python
import time
from collections import defaultdict
from typing import Dict, Set

class TemperatureAwareStorage:
    """
    温度感知存储

    自动在内存和磁盘间迁移数据
    """

    def __init__(self,
                 hot_threshold: float = 10.0,  # 每秒访问次数
                 cold_threshold: float = 0.1,
                 memory_limit: int = 1024*1024*1024):
        self.hot_threshold = hot_threshold
        self.cold_threshold = cold_threshold
        self.memory_limit = memory_limit

        # 热数据（内存）
        self.hot_data: Dict[str, Any] = {}

        # 温数据（SSD缓存）
        self.warm_data: Dict[str, Any] = {}

        # 冷数据（磁盘）
        self.cold_storage = ColdStorage()

        # 访问统计
        self.access_stats: Dict[str, 'AccessStat'] = defaultdict(
            lambda: AccessStat()
        )

        # 迁移策略
        self.migration_policy = MigrationPolicy()

    def get(self, key: str) -> Any:
        """获取数据"""
        # 更新访问统计
        self.access_stats[key].record_access()

        # 先查热区
        if key in self.hot_data:
            return self.hot_data[key]

        # 再查温区
        if key in self.warm_data:
            value = self.warm_data[key]
            # 考虑升级到热区
            self._consider_promote(key, value)
            return value

        # 从冷区加载
        value = self.cold_storage.load(key)
        self._consider_promote(key, value)
        return value

    def put(self, key: str, value: Any):
        """写入数据"""
        self.access_stats[key].record_access()

        # 默认写入热区
        self.hot_data[key] = value

        # 检查内存使用，必要时降级
        self._check_memory_pressure()

    def _consider_promote(self, key: str, value: Any):
        """考虑将数据升级到热区"""
        temp = self.access_stats[key].get_temperature()

        if temp >= self.hot_threshold and key not in self.hot_data:
            # 升级到热区
            if self._ensure_hot_space():
                self.hot_data[key] = value
                if key in self.warm_data:
                    del self.warm_data[key]

    def _check_memory_pressure(self):
        """检查内存压力并执行降级"""
        import sys
        current_memory = sys.getsizeof(self.hot_data)

        if current_memory > self.memory_limit * 0.9:
            self._evict_to_warm()

    def _evict_to_warm(self):
        """将冷数据降级到温区"""
        # 按温度排序，降级最冷的数据
        sorted_keys = sorted(
            self.hot_data.keys(),
            key=lambda k: self.access_stats[k].get_temperature()
        )

        to_evict = len(sorted_keys) // 10  # 降级10%

        for key in sorted_keys[:to_evict]:
            value = self.hot_data[key]
            self.warm_data[key] = value
            del self.hot_data[key]

    def _ensure_hot_space(self) -> bool:
        """确保热区有足够空间"""
        import sys
        current = sys.getsizeof(self.hot_data)
        if current < self.memory_limit * 0.8:
            return True

        # 需要腾出空间
        self._evict_to_warm()
        return True


class AccessStat:
    """访问统计"""

    def __init__(self):
        self.access_times = []
        self.window_size = 60  # 60秒窗口

    def record_access(self):
        now = time.time()
        self.access_times.append(now)
        self._cleanup_old(now)

    def get_temperature(self) -> float:
        now = time.time()
        self._cleanup_old(now)
        return len(self.access_times) / self.window_size

    def _cleanup_old(self, now: float):
        cutoff = now - self.window_size
        self.access_times = [t for t in self.access_times if t > cutoff]

        pass

        pass
```

## 6. 内存计算优化

### 6.1 向量化执行

**定义 17.4.12** (向量化执行)：

向量化执行使用SIMD指令一次处理多个数据：
$$\text{Output} = \text{SIMD_OP}(\vec{A}, \vec{B})$$

**定理 17.4.5** (向量化加速比)：

对于向量长度 $n$ 和SIMD宽度 $w$：
$$\text{Speedup} \approx \frac{n}{\lceil n/w \rceil} \times \frac{\text{IPC}_{SIMD}}{\text{IPC}_{scalar}}$$

```cpp
/**
 * 向量化执行引擎
 */

#include <immintrin.h>
#include <cstdint>

class VectorizedEngine {
public:
    // 使用AVX-512的向量化加法
    static void AddFloat32(const float* a, const float* b,
                          float* result, size_t n) {
        size_t i = 0;

        #ifdef __AVX512F__
        // AVX-512: 一次处理16个float
        for (; i + 16 <= n; i += 16) {
            __m512 va = _mm512_loadu_ps(&a[i]);
            __m512 vb = _mm512_loadu_ps(&b[i]);
            __m512 vr = _mm512_add_ps(va, vb);
            _mm512_storeu_ps(&result[i], vr);
        }
        #elif __AVX__
        // AVX: 一次处理8个float
        for (; i + 8 <= n; i += 8) {
            __m256 va = _mm256_loadu_ps(&a[i]);
            __m256 vb = _mm256_loadu_ps(&b[i]);
            __m256 vr = _mm256_add_ps(va, vb);
            _mm256_storeu_ps(&result[i], vr);
        }
        #elif __SSE__
        // SSE: 一次处理4个float
        for (; i + 4 <= n; i += 4) {
            __m128 va = _mm_loadu_ps(&a[i]);
            __m128 vb = _mm_loadu_ps(&b[i]);
            __m128 vr = _mm_add_ps(va, vb);
            _mm_storeu_ps(&result[i], vr);
        }
        #endif

        // 处理剩余元素
        for (; i < n; ++i) {
            result[i] = a[i] + b[i];
        }
    }

    // 向量化过滤
    static size_t FilterGreaterThan(const int32_t* data,
                                     size_t n,
                                     int32_t threshold,
                                     int32_t* result) {
        size_t count = 0;

        #ifdef __AVX2__
        __m256i thresh_vec = _mm256_set1_epi32(threshold);

        for (size_t i = 0; i + 8 <= n; i += 8) {
            __m256i vec = _mm256_loadu_si256(
                reinterpret_cast<const __m256i*>(&data[i])
            );
            __m256i cmp = _mm256_cmpgt_epi32(vec, thresh_vec);
            int mask = _mm256_movemask_epi8(cmp);

            // 提取匹配的元素
            // 简化实现，实际使用compress指令
            for (int j = 0; j < 8; ++j) {
                if (data[i + j] > threshold) {
                    result[count++] = data[i + j];
                }
            }
        }
        #else
        for (size_t i = 0; i < n; ++i) {
            if (data[i] > threshold) {
                result[count++] = data[i];
            }
        }
        #endif

        return count;
    }

    // 向量化哈希聚合
    template<typename K, typename V>
    static void HashAggregate(const K* keys, const V* values,
                              size_t n,
                              std::unordered_map<K, V>& result) {
        // 批处理以减少哈希表访问开销
        constexpr size_t BATCH_SIZE = 1024;

        for (size_t batch_start = 0; batch_start < n;
             batch_start += BATCH_SIZE) {
            size_t batch_end = std::min(batch_start + BATCH_SIZE, n);

            // 局部聚合减少竞争
            std::unordered_map<K, V> local_agg;

            for (size_t i = batch_start; i < batch_end; ++i) {
                local_agg[keys[i]] += values[i];
            }

            // 合并到全局结果
            for (const auto& [k, v] : local_agg) {
                result[k] += v;
            }
        }
    }
};
```

```cpp
/**
 * JIT编译执行
 *
 * 使用LLVM将查询计划编译为机器码

#ifdef USE_LLVM
#include <llvm/IR/IRBuilder.h>
#include <llvm/IR/LLVMContext.h>
#include <llvm/IR/Module.h>
#include <llvm/ExecutionEngine/ExecutionEngine.h>

class JITCompiler {
public:
    JITCompiler() {
        context_ = std::make_unique<llvm::LLVMContext>();
        module_ = std::make_unique<llvm::Module>("query", *context_);
        builder_ = std::make_unique<llvm::IRBuilder<>>(*context_);
    }

    // 编译表达式
    llvm::Function* CompileExpression(const Expression* expr) {
        // 创建函数
        llvm::FunctionType* func_type = llvm::FunctionType::get(
            builder_->getDoubleTy(),
            {builder_->getDoubleTy()->getPointerTo()},
            false
        );

        llvm::Function* func = llvm::Function::Create(
            func_type,
            llvm::Function::ExternalLinkage,
            "expr",
            module_.get()
        );

        // 创建基本块
        llvm::BasicBlock* bb = llvm::BasicBlock::Create(*context_, "entry", func);
        builder_->SetInsertPoint(bb);

        // 生成表达式代码
        llvm::Value* result = GenerateExprCode(expr, func->arg_begin());

        builder_->CreateRet(result);

        return func;
    }

    llvm::Value* GenerateExprCode(const Expression* expr,
                                  llvm::Value* row_ptr) {
        if (auto* col = dynamic_cast<const ColumnRef*>(expr)) {
            // 加载列值
            return builder_->CreateLoad(
                builder_->getDoubleTy(),
                builder_->CreateGEP(
                    builder_->getDoubleTy(),
                    row_ptr,
                    builder_->getInt32(col->index)
                )
            );
        }

        if (auto* bin = dynamic_cast<const BinaryExpr*>(expr)) {
            llvm::Value* left = GenerateExprCode(bin->left, row_ptr);
            llvm::Value* right = GenerateExprCode(bin->right, row_ptr);

            switch (bin->op) {
                case Operator::ADD:
                    return builder_->CreateFAdd(left, right);
                case Operator::SUB:
                    return builder_->CreateFSub(left, right);
                case Operator::MUL:
                    return builder_->CreateFMul(left, right);
                case Operator::DIV:
                    return builder_->CreateFDiv(left, right);
                default:
                    return nullptr;
            }
        }

        if (auto* constant = dynamic_cast<const Constant*>(expr)) {
            return llvm::ConstantFP::get(*context_,
                llvm::APFloat(constant->value));
        }

        return nullptr;
    }

private:
    std::unique_ptr<llvm::LLVMContext> context_;
    std::unique_ptr<llvm::Module> module_;
    std::unique_ptr<llvm::IRBuilder<>> builder_;
};
#endif
```

```c
int dictRehashStep(dict *d) {
    if (d->pauserehash == 0) {
        // 每次只迁移一个桶
        return dictRehash(d, 1);
    }
    return 0;

// 在访问时触发rehash
dictEntry *dictFind(dict *d, const void *key) {
    dictEntry *he;
    uint64_t h, idx, table;

    // 渐进式rehash
    if (dictIsRehashing(d)) {
        dictRehashStep(d);
    }

    h = dictHashKey(d, key);

    // 搜索两个表
    for (table = 0; table <= 1; table++) {
        idx = h & DICTHT_SIZE_MASK(d->ht[table].size);
        he = d->ht[table].table[idx];

        while(he) {
            if (key==he->key || dictCompareKeys(d, key, he->key))
                return he;
            he = he->next;
        }

        if (!dictIsRehashing(d)) return NULL;
    }
    return NULL;
}
```
