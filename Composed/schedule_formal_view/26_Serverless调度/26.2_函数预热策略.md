## 1 概述

### 1.1 核心洞察

函数预热是消除冷启动的主动策略，通过提前准备好可用实例来减少首次调用延迟。预热策略的核心挑战是**预测准确性**与**资源成本**的权衡。

### 1.2 预热策略分类

| 策略类型 | 描述 | 优点 | 缺点 |
|---------|------|------|------|
| **被动保持** | 延长实例存活时间 | 简单 | 资源浪费 |
| **预置并发** | 预先分配固定实例 | 可靠 | 成本高 |
| **预测预热** | 基于预测主动预热 | 高效 | 预测难 |
| **池化预热** | 维护通用实例池 | 灵活 | 初始化开销 |

```python
class KeepAliveManager:
    """Keep-Alive实例管理器"""

    def __init__(self, default_timeout: float = 300.0):
        self.default_timeout = default_timeout
        self.instances: Dict[str, Instance] = {}
        self.timers: Dict[str, Timer] = {}

    def get_or_create(self, function_id: str) -> Instance:
        """获取或创建实例"""
        if function_id in self.instances:
            # 复用现有实例，重置定时器
            self._reset_timer(function_id)
            return self.instances[function_id]
        else:
            # 创建新实例
            instance = Instance.create(function_id)
            self.instances[function_id] = instance
            self._start_timer(function_id)
            return instance

    def _start_timer(self, function_id: str):
        """启动Keep-Alive定时器"""
        timeout = self._get_timeout(function_id)
        timer = Timer(timeout, lambda: self._on_timeout(function_id))
        timer.start()
        self.timers[function_id] = timer

    def _reset_timer(self, function_id: str):
        """重置定时器"""
        if function_id in self.timers:
            self.timers[function_id].cancel()
        self._start_timer(function_id)

    def _on_timeout(self, function_id: str):
        """超时回调"""
        if function_id in self.instances:
            self.instances[function_id].terminate()
            del self.instances[function_id]
            del self.timers[function_id]

    def _get_timeout(self, function_id: str) -> float:
        """获取自适应超时时间"""
        # 可基于历史调用模式动态调整
        return self.default_timeout
```

```python
import numpy as np
from sklearn.ensemble import RandomForestRegressor

class PredictiveWarmer:
    """预测预热调度器"""

    def __init__(self):
        self.model = RandomForestRegressor()
        self.history: Dict[str, List[CallRecord]] = defaultdict(list)

    def record_call(self, function_id: str, timestamp: datetime):
        """记录调用"""
        self.history[function_id].append(CallRecord(timestamp))

    def predict_load(self, function_id: str, future_window: timedelta) -> int:
        """预测未来负载"""
        features = self._extract_features(function_id, future_window)
        predicted_calls = self.model.predict([features])[0]
        return int(np.ceil(predicted_calls))

    def _extract_features(self, function_id: str, window: timedelta) -> List[float]:
        """提取预测特征"""
        now = datetime.now()
        history = self.history[function_id]

        features = [
            # 时间特征
            now.hour,
            now.weekday(),
            now.day,
            # 历史特征
            self._calls_in_window(history, timedelta(hours=1)),
            self._calls_in_window(history, timedelta(days=1)),
            self._calls_in_window(history, timedelta(weeks=1)),
            # 趋势特征
            self._trend(history),
        ]
        return features

    def schedule_warmup(self, function_id: str):
        """调度预热"""
        predicted_load = self.predict_load(function_id, timedelta(minutes=5))
        current_instances = self.get_warm_instances(function_id)

        if predicted_load > current_instances:
            # 需要预热更多实例
            warmup_count = predicted_load - current_instances
            self._warmup_instances(function_id, warmup_count)

    def _warmup_instances(self, function_id: str, count: int):
        """执行预热"""
        for _ in range(count):
            instance = Instance.create(function_id)
            instance.initialize()  # 预初始化
            self.warm_pool.add(function_id, instance)
```

### 4.4 预测预温热身调度器实现（Python + Rust）

以下实现一个完整的、基于时间序列预测的函数预热调度器，包含Python主控逻辑和Rust高性能核心。

#### Python 主控层

```python
# predictive_warmer.py
import asyncio
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Dict, List
import numpy as np
from collections import deque, defaultdict

@dataclass
class FunctionMetrics:
    function_id: str
    call_times: deque  # 最近1000次调用时间戳
    avg_exec_time: float
    cold_start_latency: float
    warm_start_latency: float

class PredictivePreWarmer:
    """
    基于EWMA和周期性模式的预测预热调度器
    """

    def __init__(self, lookback_window: int = 3600, forecast_horizon: int = 300):
        self.lookback_window = lookback_window    # 历史观察窗口（秒）
        self.forecast_horizon = forecast_horizon  # 预测窗口（秒）
        self.metrics: Dict[str, FunctionMetrics] = {}
        self.warm_instances: Dict[str, int] = defaultdict(int)
        self.cold_penalty = 10.0   # 冷启动惩罚系数
        self.warm_cost = 1.0       # 每实例每秒预热成本

    def register_function(self, function_id: str, cold_latency: float, warm_latency: float):
        self.metrics[function_id] = FunctionMetrics(
            function_id=function_id,
            call_times=deque(maxlen=10000),
            avg_exec_time=0.1,
            cold_start_latency=cold_latency,
            warm_start_latency=warm_latency,
        )

    def record_invocation(self, function_id: str, timestamp: datetime = None):
        if timestamp is None:
            timestamp = datetime.utcnow()
        self.metrics[function_id].call_times.append(timestamp)

    def predict_concurrent_load(self, function_id: str) -> int:
        """
        预测未来forecast_horizon秒内需要的并发实例数
        方法：基于历史同期数据的加权平均
        """
        metrics = self.metrics[function_id]
        now = datetime.utcnow()
        horizon = timedelta(seconds=self.forecast_horizon)

        # 收集历史同期数据（过去7天同一时间段）
        predictions = []
        weights = []

        for day_offset in range(1, 8):
            past_start = now - timedelta(days=day_offset)
            past_end = past_start + horizon

            # 计算该时间段内的调用次数
            count = sum(
                1 for t in metrics.call_times
                if past_start <= t <= past_end
            )
            # 计算该时间段的峰值并发（简化为调用数，实际应除以执行时间）
            predictions.append(count)
            weights.append(0.5 ** (day_offset - 1))  # 越近权重越高

        if not predictions:
            return 1

        weighted_avg = np.average(predictions, weights=weights)
        # 考虑执行时间得到并发需求
        concurrent_needed = int(np.ceil(
            weighted_avg * metrics.avg_exec_time / self.forecast_horizon
        ))

        # 加入安全余量（基于标准差）
        std = np.std(predictions) if len(predictions) > 1 else 0
        safety_margin = int(np.ceil(std * metrics.avg_exec_time / self.forecast_horizon))

        return max(1, concurrent_needed + safety_margin)

    def compute_optimal_warm_pool(self, function_id: str) -> dict:
        """
        基于成本-延迟权衡计算最优预热实例数
        """
        metrics = self.metrics[function_id]
        predicted = self.predict_concurrent_load(function_id)

        best_cost = float('inf')
        best_n = 1

        for n in range(1, predicted * 3 + 1):
            # 假设请求服从泊松分布
            lambda_rate = predicted / metrics.avg_exec_time
            p_cold = 1 - sum(
                (lambda_rate * metrics.avg_exec_time) ** k / math.factorial(k)
                * math.exp(-lambda_rate * metrics.avg_exec_time)
                for k in range(n + 1)
            )

            cost = (
                n * self.warm_cost * self.forecast_horizon +
                p_cold * self.cold_penalty * lambda_rate * self.forecast_horizon
            )

            if cost < best_cost:
                best_cost = cost
                best_n = n

        return {
            'function_id': function_id,
            'predicted_concurrent': predicted,
            'optimal_warm_instances': best_n,
            'estimated_cost': best_cost,
        }

    async def run(self, interval: int = 30):
        """主循环：定期评估并调整预热池"""
        while True:
            for function_id in self.metrics:
                decision = self.compute_optimal_warm_pool(function_id)
                current = self.warm_instances[function_id]
                target = decision['optimal_warm_instances']

                if target > current:
                    await self._scale_up(function_id, target - current)
                elif target < current:
                    await self._scale_down(function_id, current - target)

                self.warm_instances[function_id] = target

            await asyncio.sleep(interval)

    async def _scale_up(self, function_id: str, count: int):
        """异步预热实例"""
        print(f"[WARMUP] Scaling up {function_id} by {count} instances")
        # 实际调用容器/VM管理API

    async def _scale_down(self, function_id: str, count: int):
        """释放多余实例"""
        print(f"[WARMUP] Scaling down {function_id} by {count} instances")
```

#### Rust 高性能核心

```rust
// src/lib.rs
use std::collections::{HashMap, VecDeque};
use std::sync::{Arc, Mutex};
use pyo3::prelude::*;

/// Rust实现的轻量级EWMA预测器
/// 用于Python主控层的高频调用
#[pyclass]
struct RustPredictor {
    histories: HashMap<String, VecDeque<f64>>,
    alpha: f64,  // EWMA平滑因子
}

#[pymethods]
impl RustPredictor {
    #[new]
    fn new(alpha: f64) -> Self {
        RustPredictor {
            histories: HashMap::new(),
            alpha,
        }
    }

    fn record(&mut self, function_id: String, timestamp: f64) {
        self.histories
            .entry(function_id)
            .or_insert_with(|| VecDeque::with_capacity(10000))
            .push_back(timestamp);
    }

    fn predict_rate(&self, function_id: &str, window_seconds: f64) -> PyResult<f64> {
        let history = self.histories.get(function_id)
            .ok_or_else(|| pyo3::exceptions::PyKeyError::new_err(
                "Function not found"
            ))?;

        if history.len() < 2 {
            return Ok(0.0);
        }

        let now = *history.back().unwrap();
        let window_start = now - window_seconds;

        // 计算窗口内调用次数
        let count = history.iter()
            .filter(|&&t| t >= window_start)
            .count() as f64;

        // EWMA预测
        let rate = count / window_seconds;
        Ok(rate)
    }

    fn predict_peak_concurrent(
        &self,
        function_id: &str,
        forecast_seconds: f64,
        exec_time: f64,
    ) -> PyResult<usize> {
        let rate = self.predict_rate(function_id, 3600.0)?;
        let predicted_calls = rate * forecast_seconds;
        let concurrent = (predicted_calls * exec_time / forecast_seconds).ceil() as usize;
        Ok(concurrent.max(1))
    }

    fn batch_optimal_warm_pool(
        &self,
        functions: Vec<String>,
        forecast_seconds: f64,
        exec_times: Vec<f64>,
        warm_costs: Vec<f64>,
        cold_penalties: Vec<f64>,
    ) -> PyResult<Vec<usize>> {
        let results: Vec<usize> = functions.iter().enumerate().map(|(i, fid)| {
            let lambda_rate = match self.predict_rate(fid, 3600.0) {
                Ok(r) => r,
                Err(_) => return 1usize,
            };

            let exec_time = exec_times[i];
            let warm_cost = warm_costs[i];
            let cold_penalty = cold_penalties[i];
            let mean_concurrent = lambda_rate * exec_time;

            let mut best_n = 1;
            let mut best_cost = f64::INFINITY;

            for n in 1..=(mean_concurrent as usize * 3).max(10) {
                // 使用正态近似计算冷启动概率
                let sigma = mean_concurrent.sqrt();
                let z = (n as f64 - mean_concurrent) / sigma.max(1e-6);

                // 标准正态CDF近似（Abramowitz & Stegun）
                let p_cold = 1.0 - normal_cdf(z);

                let cost = n as f64 * warm_cost * forecast_seconds
                    + p_cold * cold_penalty * lambda_rate * forecast_seconds;

                if cost < best_cost {
                    best_cost = cost;
                    best_n = n;
                }
            }

            best_n
        }).collect();

        Ok(results)
    }
}

fn normal_cdf(x: f64) -> f64 {
    // Abramowitz & Stegun 近似
    let b1 = 0.319381530;
    let b2 = -0.356563782;
    let b3 = 1.781477937;
    let b4 = -1.821255978;
    let b5 = 1.330274429;
    let p = 0.2316419;
    let c = 0.39894228;

    if x >= 0.0 {
        let t = 1.0 / (1.0 + p * x);
        1.0 - c * (-x * x / 2.0).exp() * t *
            (t * (t * (t * (t * b5 + b4) + b3) + b2) + b1)
    } else {
        1.0 - normal_cdf(-x)
    }
}

#[pymodule]
fn serverless_warmer_core(_py: Python, m: &PyModule) -> PyResult<()> {
    m.add_class::<RustPredictor>()?;
    Ok(())
}
```

class TimeSeriesPredictor:

    def train(self, function_id: str, history: pd.DataFrame):

                    time=row['ds'] - timedelta(minutes=5),  # 提前5分钟
                    target_instances=int(np.ceil(row['yhat_upper'])),
                ))

        return schedule

```python
class InstancePoolManager:
    """多级实例池管理器"""

    def __init__(self):
        self.hot_pool: Dict[str, List[Instance]] = defaultdict(list)
        self.warm_pool: Dict[str, List[Instance]] = defaultdict(list)
        self.cold_pool: Dict[str, List[Instance]] = defaultdict(list)

        # 池大小配置
        self.config = PoolConfig(
            hot_pool_size=5,
            warm_pool_size=20,
            cold_pool_size=100,
        )

    def get_instance(self, function_id: str) -> Instance:
        """获取实例（按优先级）"""
        # 1. 尝试热池
        if self.hot_pool[function_id]:
            return self.hot_pool[function_id].pop()

        # 2. 尝试温池
        if self.warm_pool[function_id]:
            instance = self.warm_pool[function_id].pop()
            instance.complete_init()  # 完成初始化
            return instance

        # 3. 尝试冷池
        if self.cold_pool[function_id]:
            instance = self.cold_pool[function_id].pop()
            instance.full_init()  # 完整初始化
            return instance

        # 4. 创建新实例
        return Instance.create(function_id)

    def return_instance(self, function_id: str, instance: Instance):
        """归还实例到池"""
        if len(self.hot_pool[function_id]) < self.config.hot_pool_size:
            self.hot_pool[function_id].append(instance)
        else:
            instance.partial_cleanup()  # 部分清理
            if len(self.warm_pool[function_id]) < self.config.warm_pool_size:
                self.warm_pool[function_id].append(instance)
            else:
                instance.terminate()  # 释放

    def maintain_pools(self):
        """池维护任务"""
        for function_id in self.get_active_functions():
            # 预测负载
            predicted_load = self.predictor.predict(function_id)

            # 调整池大小
            self._adjust_pool_size(function_id, predicted_load)

    def _adjust_pool_size(self, function_id: str, target: int):
        """调整池大小"""
        current = len(self.hot_pool[function_id])

        if target > current:
            # 从温池提升到热池
            to_promote = min(target - current, len(self.warm_pool[function_id]))
            for _ in range(to_promote):
                instance = self.warm_pool[function_id].pop()
                instance.complete_init()
                self.hot_pool[function_id].append(instance)
```

## 7 形式化模型

### 7.1 预热决策模型

```text
预热决策问题:

给定:
  λ(t): 时刻t的预测请求率
  C_warm: 单实例预热成本/时间
  C_cold: 冷启动惩罚成本
  C_idle: 空闲实例成本/时间

决策变量:
  N(t): 时刻t的预热实例数

目标:
  minimize ∫[C_idle × max(0, N(t) - λ(t)) + C_cold × max(0, λ(t) - N(t))] dt

约束:
  N(t) ≥ N_min  (最小实例)
  N(t) ≤ N_max  (最大实例)
  |N(t+1) - N(t)| ≤ R_scale  (伸缩速率限制)
```

### 7.2 最优预热数量

```text
定理: 最优预热实例数

在平稳负载假设下:
  N* = λ + z_{1-α} × σ

其中:
  λ = 平均请求率
  σ = 请求率标准差
  z_{1-α} = 正态分布分位数 (SLA对应)

证明:
  设冷启动概率 P_cold = P(Requests > N)
  对于正态分布: P_cold = 1 - Φ((N - λ)/σ)

  要满足 P_cold ≤ α:
  1 - Φ((N - λ)/σ) ≤ α
  (N - λ)/σ ≥ z_{1-α}
  N ≥ λ + z_{1-α} × σ
```
