## 2 VFS调度

### 2.1 路径查找优化

**RCU路径查找**：

```c
// 无锁路径查找
struct dentry *lookup_fast(struct nameidata *nd,
                           struct inode *inode,
                           unsigned *seqp) {
    struct dentry *dentry;

    // 使用RCU读取dentry缓存
    rcu_read_lock();
    dentry = __d_lookup_rcu(nd->path.dentry, &nd->last, seqp);
    if (unlikely(!dentry)) {
        rcu_read_unlock();
        return NULL;
    }

    // 验证inode序列号
    if (read_seqcount_retry(&dentry->d_seq, *seqp)) {
        rcu_read_unlock();
        return NULL;
    }

    rcu_read_unlock();
    return dentry;
}
```

### 2.2 dentry缓存调度

**LRU管理**：

```c
// dentry缓存回收
void prune_dcache_sb(struct super_block *sb, int nr_to_scan) {
    struct dentry *dentry;

    while (nr_to_scan-- && !list_empty(&sb->s_dentry_lru)) {
        dentry = list_first_entry(&sb->s_dentry_lru,
                                   struct dentry, d_lru);

        // 检查引用计数
        if (dentry->d_lockref.count == 0) {
            // 安全回收
            __dentry_kill(dentry);
        }
    }
}
```

## 3 页缓存管理

### 3.1 页缓存结构

**radix树索引**：

```c
// 页缓存地址空间
struct address_space {
    struct inode *host;
    struct radix_tree_root page_tree;  // radix树索引页
    struct list_head lru;               // LRU链表
    unsigned long nrpages;
    const struct address_space_operations *a_ops;
};

// 查找页缓存
struct page *find_get_page(struct address_space *mapping,
                           pgoff_t offset) {
    struct page *page;

    rcu_read_lock();
    page = radix_tree_lookup(&mapping->page_tree, offset);
    if (page && !page_cache_get_speculative(page)) {
        page = NULL;
    }
    rcu_read_unlock();

    return page;
}
```

### 3.2 预读策略

**自适应预读**：

```c
// 预读窗口调整
void page_cache_async_readahead(struct address_space *mapping,
                                 struct file_ra_state *ra,
                                 struct file *filp,
                                 struct page *page,
                                 pgoff_t offset) {
    // 检测顺序访问
    if (offset == ra->start + ra->size) {
        // 顺序访问，增加预读窗口
        ra->size = min(ra->size * 2, ra->async_size);
    } else {
        // 随机访问，禁用预读
        ra->size = 0;
    }

    // 发起预读
    force_page_cache_readahead(mapping, filp, offset, ra->size);
}
```

### 3.4 缓存回收

**LRU/2Q算法**：

```
活跃链表（Active）：最近访问的页
非活跃链表（Inactive）：较少访问的页

回收策略：
1. 优先回收非活跃链表
2. 访问非活跃页时移到活跃链表
3. 活跃链表满时，尾部移到非活跃链表
```

### 4.2 多队列调度

**blk-mq架构**：

```
应用程序
    ↓ submit_bio
块层
    ↓
软件队列（每CPU）
    ↓
硬件队列（映射到设备队列）
    ↓
设备驱动
```

```c
// blk-mq提交
void blk_mq_submit_bio(struct bio *bio) {
    struct request_queue *q = bio->bi_disk->queue;
    struct blk_mq_hw_ctx *hctx;
    struct request *rq;

    // 获取硬件队列上下文
    hctx = blk_mq_map_queue(q, bio->bi_opf, &bio->bi_disk->part0);

    // 分配请求
    rq = blk_mq_alloc_request(hctx, bio->bi_opf, GFP_NOIO);

    // 初始化请求
    blk_mq_bio_to_request(rq, bio);

    // 插入队列
    blk_mq_insert_request(rq, false, true, true);
}
```

### 4.3 IO合并与排序

**电梯算法**：

```c
// 电梯排序
void elevator_add_req(struct request_queue *q, struct request *rq) {
    struct elevator_queue *eq = q->elevator;
    struct request *pos;

    // 按扇区号排序插入
    list_for_each_entry_reverse(pos, &eq->queue_head, queuelist) {
        if (pos->sector <= rq->sector)
            break;
    }

    list_add(&rq->queuelist, &pos->queuelist);
}
```

## 5 形式化定义

**文件系统操作延迟**：

$$
T_{fs} = T_{vfs} + T_{cache} + T_{disk}
$$

其中：

- $T_{vfs}$：VFS层处理时间
- $T_{cache}$：缓存查找/更新时间
- $T_{disk}$：磁盘IO时间（缓存命中时为0）

**缓存命中率**：

$$
H_{cache} = \frac{N_{cache\_hit}}{N_{total\_access}}
$$

**预读效果**：

$$
E_{readahead} = \frac{T_{without} - T_{with}}{T_{without}}
$$

## 6 Linux内核实现分析

**文件读取流程**：

```c
// 通用文件读取
ssize_t generic_file_read_iter(struct kiocb *iocb, struct iov_iter *iter) {
    struct file *file = iocb->ki_filp;
    struct address_space *mapping = file->f_mapping;
    struct inode *inode = mapping->host;

    // 1. 检查页缓存
    page = find_get_page(mapping, index);
    if (page) {
        // 缓存命中
        copy_page_to_iter(page, offset, bytes, iter);
        put_page(page);
        return bytes;
    }

    // 2. 缓存未命中，启动IO
    page = page_cache_alloc(mapping);
    add_to_page_cache_lru(page, mapping, index);

    // 3. 提交读请求
    ret = mapping->a_ops->readpage(file, page);

    // 4. 等待IO完成
    wait_on_page_locked(page);

    // 5. 拷贝数据
    copy_page_to_iter(page, offset, bytes, iter);

    return bytes;
}
```
