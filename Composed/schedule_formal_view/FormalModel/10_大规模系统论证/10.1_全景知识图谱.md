            params.insert(key, value);
        }

        Ok(())
    }

        &self,

            "MATCH (a:{} {{id: $from_id}}), (b:{} {{id: $to_id}}) \

        params.insert("from_id".to_string(), from_id.to_string());

        self.graph.execute(query(query).params(params)).await?;
