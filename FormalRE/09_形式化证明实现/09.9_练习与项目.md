  | BNot : bexp -> bexp
  | BAnd : bexp -> bexp -> bexp.

(* 求值函数 *)
Fixpoint beval (e : bexp) : bool :=
  match e with
  | BTrue => true
  | BFalse => false
  | BNot e' => negb (beval e')

  end.
