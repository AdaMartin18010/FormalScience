Definition equivalent (S1 S2 : Type)
                      (accepts1 : S1 -> input -> Prop)
                      (accepts2 : S2 -> input -> Prop) :=
  (forall m1 : S1, exists m2 : S2,
     forall i, accepts1 m1 i <-> accepts2 m2 i) /\
